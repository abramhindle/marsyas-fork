48 void MedianFilter::myProcess(realvec& inVec, realvec& outVec)
49 {
50  // Each element in the input vector is replaced by the median of
51  // the elements that fall in a window surrounding the element, defined as:
52  // element index + [0,1.. N-1] - floor(N/2) with N, the size of the window.
53 
54  // Define moving window: once inserted,
55  // the elements in the moving window are automatically sorted
56  mrs_natural N = WindowSize_;
57  multimap<mrs_real,mrs_natural,less<mrs_real> > theWindow;
58  typedef multimap<mrs_real,mrs_natural,less<mrs_real> >::iterator iter;
59  typedef pair<mrs_real,mrs_natural> element;
60 
61  // M = floor(N/2)
62  mrs_natural M = (mrs_natural)floor((mrs_real)N/2.);
63 
64  // Initialize the moving window
65  // For i=0, the window = [v(0), v(0),.. v(0),v(1),.. v(N-M)]
66  for (int p=-M; p<=0; p++)
67  theWindow.insert(element(inVec(0),p));
68  for (int p=1; p<=N-M-1; p++)
69  theWindow.insert(element(inVec(p),p));
70 
71  mrs_natural I = inVec.getSize();
72  for (int i=0; i<I; ++i)
73  {
74  // Define to-be-inserted element
75  element theNewOne(inVec(min(i+N-M,I-1)),i+N-M);
76 
77  // Search for median, insert and delete position
78  int theOffset = -M;
79 
80  iter theMedian, theToDelete, theToInsert = theWindow.begin();
81  for (iter theIter=theWindow.begin(); theIter!=theWindow.end(); theIter++)
82  {
83  if (theOffset == 0)
84  theMedian = theIter;
85  if (theIter->second == i-M)
86  // Index i-M = first inserted element (oldest)
87  theToDelete = theIter;
88  if (theNewOne.first > theIter->first)
89  {
90  // Update theToInsert as long as theIter
91  // is smaller than the to-be-inserted element
92  theToInsert = theIter;
93  theToInsert++;
94  }
95  theOffset++;
96  }
97 
98  if (theToInsert == theToDelete)
99  theToInsert++;
100 
101  outVec(i) = theMedian->first;
102  theWindow.erase(theToDelete);
103  theWindow.insert(theToInsert,theNewOne);
104  }
105 }
