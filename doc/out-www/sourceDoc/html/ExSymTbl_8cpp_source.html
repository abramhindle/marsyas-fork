122 bool
123 ExRecord::params_compare(std::string a, std::string b)
124 {
125  int p1=0;
126  int p2=0;
127  int len1=a.length();
128  int len2=b.length();
129  int start=1;
130 
131  while (p1<len1&&p2<len2) {
132  if (a[p1]!=b[p2]) {
133  if (a[p1]==','&&b[p2]=='|') {
134  while (p2<len2) {
135  if (b[p2]==',') break;
136  if (b[p2]==')') return false;
137  p2++;
138  }
139  p1++;
140  p2++;
141  }
142  else if (a[p1]==')'&&b[p2]=='|') {
143  while (p2<len2) {
144  if (b[p2]==',') return false;
145  if (b[p2]==')') return true;
146  p2++;
147  }
148  return false;
149  }
150  else {
151  while(p2<len2) {
152  if (b[p2]=='|') break;
153  if (b[p2]==','||b[p2]==')') return false;
154  p2++;
155  }
156  p1=start; p2++;
157  }
158  }
159  else {
160  if (a[p1]==',') start=p1+1;
161  p1++; p2++;
162  }
163  if (a[p1]==')'&&b[p2]==')') return true;
164  }
165  return false;
166 }

201 ExRecord*
202 ExRecord::find_sym(std::string path)
203 {
204  // get leading name in group: a|b|c.d|e|f.g|h|i => a|b|c d|e|f.g|h|i
205  std::string ghd, gtl; split_on(path,'.',ghd,gtl);
206  if (gtl==""&&ghd[0]!='(') {
207  split_on(ghd,'(',ghd,gtl,true);
208  }
209  if (gtl!="") { // then ghd is the variable name
210  // get leading name in name: a|b|c => a b|c
211  std::map<std::string, ExRecord*>::iterator gi=syms_.find(ghd);
212  if (gi==syms_.end()) {
213  std::map<std::string,std::string>::iterator it = syms_aliases_.find(ghd);
214  // a possible error condition since the path given diverges from the possible paths
215  if (it==syms_aliases_.end())
216  return NULL;
217  else
218  return syms_[it->second]->find_sym(gtl);
219  }
220 
221  return (gi->second)->find_sym(gtl);
222  }
223  if (path[0]=='(') {
224  ExRecord* answer=NULL;
225  std::map<std::string,ExRecord*>::iterator is=syms_.begin();
226  while (is!=syms_.end()&&!answer) {
227  if (params_compare((is->first),path)) { answer=is->second; }
228  else is++;
229  }
230  if (!answer) {
231  std::map<std::string,std::string>::iterator it=syms_aliases_.begin();
232  while (it!=syms_aliases_.end()&&!answer) {
233  if (params_compare((it->first),path))
234  answer=syms_[it->second];
235  else
236  ++it;
237  }
238  }
239  return answer;
240  }
241  std::map<std::string,ExRecord*>::iterator gi=syms_.find(ghd);
242  if (gi==syms_.end()) {
243  // name not found, so see if it is an alias
244  std::map<std::string,std::string>::iterator ni=syms_aliases_.find(ghd);
245  if (ni==syms_aliases_.end()) { return NULL; }
246  return syms_[ni->second];
247  }
248  return (gi->second);
249 }

263 void
264 ExRecord::addRecord(std::string path, ExRecord* sym)
265 {
266  // get leading name in group: a|b|c.d|e|f.g|h|i => a|b|c d|e|f.g|h|i
267  std::string ghd; std::string gtl; split_on(path,'.',ghd,gtl);
268  if (gtl!="") { // then ghd is the variable name
269  // now split on '|' symbol, get leading name in name: a|b|c => a b|c
270  std::string nhd, ntl; split_on(ghd,'|',nhd,ntl);
271  std::string name_=nhd;
272  std::map<std::string, ExRecord*>::iterator gi=syms_.find(name_);
273  ExRecord* syms=NULL;
274  // for first split, if name doesn't exist, add it as a record
275  if (gi==syms_.end()) { syms=new ExRecord(T_LIB); syms_[name_]=syms; syms->inc_ref(); }
276  else { syms=(gi->second); }
277 
278  // the rest of the names are aliases
279  addAliases(ntl,name_);
280  syms->addRecord(gtl,sym);
281  }
282  else {
283  // discover parameters
284  std::string params;
285  if (ghd[0]!='(') split_on(ghd,'(',ghd,params,true);
286  // get leading name in name: a|b|c => a b|c, becomes the master name
287  std::string nhd, ntl; split_on(ghd,'|',nhd,ntl);
288  std::map<std::string, ExRecord*>::iterator gi=syms_.find(nhd);
289  if (gi==syms_.end()) { // not found, so must be a new name
290  if (params!="") {
291  ExRecord* e=new ExRecord(T_FUN); e->inc_ref();
292  syms_[nhd]=e;
293  e->addRecord(params,sym);
294  }
295  else {
296  syms_[nhd]=sym;
297  sym->inc_ref();
298  }
299  }
300  else if (params!="") {
301  (gi->second)->addRecord(params,sym);
302  }
303  else {
304  MRSWARN("ExRecord::addRecord '"+nhd+"' already refers to a symbol");
305  sym->deref();
306  return;
307  }
308  addAliases(ntl,nhd);
309  }

311 ExRecord*
312 ExRecord::rmvRecord(std::string path)
313 {
314  // get leading name in group: a|b|c.d|e|f.g|h|i => a|b|c d|e|f.g|h|i
315  std::string ghd, gtl; split_on(path,'.',ghd,gtl);
316  std::map<std::string, ExRecord*>::iterator rec_i;
317  std::map<std::string,std::string>::iterator alias_i;
318 
319  if (gtl==""&&ghd[0]!='(') {
320  split_on(ghd,'(',ghd,gtl,true);
321  }
322  rec_i=syms_.find(ghd);
323  if (rec_i==syms_.end()) {
324  alias_i=syms_aliases_.find(ghd);
325  // a possible error condition since the path given diverges from the possible paths
326  if (alias_i==syms_aliases_.end()) { return NULL; }
327  ghd=alias_i->second;
328  rec_i=syms_.find(ghd); // let's assume that gi is valid
329  }
330  ExRecord* r;
331  bool del_rec=false;
332  if (gtl!="") r=(rec_i->second)->rmvRecord(gtl);
333  else { r=rec_i->second; del_rec=true; }
334  if (r!=NULL) {
335  std::vector<std::string> names_;
336  // eliminate aliases to r by first getting their names
337  for (alias_i=syms_aliases_.begin();alias_i!=syms_aliases_.end();++alias_i) {
338  if (alias_i->second==ghd) names_.push_back(alias_i->first);
339  }
340  if (names_.size()>0) {
341  // delete aliases from syms_aliases_
342  std::vector<std::string>::iterator nmi;
343  for (nmi=names_.begin();nmi!=names_.end();++nmi) {
344  syms_aliases_.erase(*nmi);
345  }
346  }
347  // eliminate the record from syms_
348  syms_.erase(rec_i);
349  if (del_rec&&r->size()>0) r->deref();
350  }
351  return r;
352 }

364 void
365 ExRecord::setValue(ExVal& v, std::string path, int elem_pos)
366 {
367  if (path!="") {
368  // get leading name in group: a|b|c.d|e|f.g|h|i => a|b|c d|e|f.g|h|i
369  std::string name; split_on(path,'.',name,path);
370  if (path!="") { // then ghd is the variable name
371  // get leading name in name: a|b|c => a b|c
372  std::map<std::string, ExRecord*>::iterator iter=syms_.find(name);
373  ExRecord* syms=NULL;
374  if (iter==syms_.end()) { syms=new ExRecord(); syms_[name]=syms; syms->inc_ref(); }
375  else { syms=iter->second; }
376  syms->setValue(v,path);
377  return;
378  }
379  else {
380  // get leading name in name: a|b|c => a b|c
381  std::map<std::string, ExRecord*>::iterator iter=syms_.find(name);
382  if (iter==syms_.end()) { // not in table, so add it
383  ExRecord* r=new ExRecord(T_VAR,name,v,false); r->inc_ref();
384  syms_[name]=r;
385  return;
386  }
387  else {
388  (iter->second)->setValue(v);
389  return;
390  }
391  }
392  }
393  if (getKind()!=T_VAR) {
394  MRSWARN("ExRecord::setValue Attempting assignment to non-variable");
395  }
396  else if (elem_pos>=0) {
397  if (getElemType()!=v.getType()) {
398  MRSWARN("ExRecord::setValue Type mismatch in assignment of element: "+getElemType()+" << "+v.getType());
399  }
400  else value_.setSeqElem(elem_pos, v);
401  }
402  else if (getType()!=v.getType()) {
403  MRSWARN("ExRecord::setValue Type mismatch in assignment: "+getType()+" << "+v.getType());
404  } else value_=v;
405 }
