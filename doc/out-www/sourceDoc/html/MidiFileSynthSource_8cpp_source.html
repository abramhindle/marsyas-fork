96 void
97 MidiFileSynthSource::myUpdate(MarControlPtr sender)
98 {
99  (void) sender; //suppress warning of unused parameter(s)
100  MRSDIAG("MidiFileSynthSource::myUpdate");
101 
102  MATLAB_PUT(ctrl_winSize_->to<mrs_natural>(), "winSize");
103  MATLAB_PUT(inSamples_, "hopSize");
104 
105  mrs_bool sigNewTextWin = ctrl_sigNewTextWin_->to<mrs_bool>();
106  MATLAB_PUT((mrs_natural)sigNewTextWin, "sigNewTextWin");
107  //in case texture windows are set from the outside...
108  if(!sigNewTextWin)
109  {
110  if(ctrl_newTextWin_->isTrue())
111  {
112  //get number of playing notes in texture window
113  mrs_natural numActiveNotes;
114  MATLAB_GET("numActiveNotes", numActiveNotes);
115  ctrl_numActiveNotes_->setValue(numActiveNotes);
116 
117  MATLAB_EVAL("textWinStart = textWinEnd+1;textWinEnd = endPos2;");
118 
119  //reset MATLAB
120  //MATLAB_EVAL("numActiveNotes = 0;");
121 
122  //reset control
123  ctrl_newTextWin_->setValue(false, NOUPDATE);
124  }
125  }
126 
127  if(filename_ != ctrl_filename_->to<mrs_string>())
128  {
129  filename_ = ctrl_filename_->to<mrs_string>();
130  MATLAB_PUT(filename_, "filename");
131  MATLAB_PUT(israte_, "fs");
132  MATLAB_PUT(ctrl_start_->to<mrs_real>(), "startSeg");
133  MATLAB_PUT(ctrl_end_->to<mrs_real>(), "endSeg");
134 
135  //Synthesize MIDI file to audio
136  MATLAB_EVAL("synthetizeMIDI;");
137 
138  MATLAB_GET("numChannels", nChannels_);
139  ctrl_nChannels_->setValue(nChannels_, NOUPDATE);
140 
141  MATLAB_GET("audioLength", size_); //in samples
142  ctrl_size_->setValue(size_, NOUPDATE);
143 
144  ctrl_pos_->setValue(0, NOUPDATE);
145  if(size_>0)
146  ctrl_hasData_->setValue(true, NOUPDATE);
147  else
148  ctrl_hasData_->setValue(false, NOUPDATE);
149  }
150 
151  ctrl_onSamples_->setValue(ctrl_inSamples_, NOUPDATE);
152  ctrl_onObservations_->setValue(1, NOUPDATE);
153  ctrl_osrate_->setValue(ctrl_israte_, NOUPDATE);
154  ctrl_onObsNames_->setValue("MIDIaudio", NOUPDATE);
155 }

157 void
158 MidiFileSynthSource::myProcess(realvec& in, realvec &out)
159 {
160  (void) in;
161  if(size_ == 0)
162  return;
163 
164  mrs_natural pos = ctrl_pos_->to<mrs_natural>();
165 
166  //detect end of file
167  if(pos > size_)
168  {
169  out.setval(0.0);//fill with silence
170 
171  if(!ctrl_hasData_->isTrue())
172  return;
173 
174  ctrl_hasData_->setValue(false);
175 
176  //flush last texture window
177  if(ctrl_sigNewTextWin_->isTrue())
178  {
179  mrs_natural numActiveNotes;
180  MATLAB_GET("numActiveNotes", numActiveNotes);
181  ctrl_numActiveNotes_->setValue(numActiveNotes);
182  ctrl_newTextWin_->setValue(true, NOUPDATE);
183  }
184 
185  return;
186  }
187 
188  MATLAB_PUT(pos+1, "pos");//MATLAB uses indexes in the range [1,...]
189  MATLAB_EVAL("computeAudioFrame;");
190  MATLAB_GET("audioFrame", out);
191 
192  ctrl_pos_->setValue(pos+onSamples_);
193 
194  //check if this audio frame is the last one in the current texture window:
195  //if it is, then signal a new texture window
196  if(ctrl_sigNewTextWin_->isTrue())
197  {
198  mrs_natural newTextWin = 0;
199  MATLAB_GET("newTextWin", newTextWin);
200  if(newTextWin!=0)//if true...
201  {
202  mrs_natural numActiveNotes;
203  MATLAB_GET("numActiveNotes", numActiveNotes);
204  ctrl_numActiveNotes_->setValue(numActiveNotes);
205  //MATLAB_EVAL("numActiveNotes = 0;"); //reset
206  }
207  ctrl_newTextWin_->setValue(newTextWin!=0, NOUPDATE);
208  }
209 }
