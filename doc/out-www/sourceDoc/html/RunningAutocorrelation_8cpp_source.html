76 mrs_string prefixObservationNamesWithAutocorrelationUnfoldingPrefixes_(

78  mrs_bool doNotNormalizeForLag0, mrs_natural maxLag)
79 {
80  vector<mrs_string> inObsNames = obsNamesSplit(inObservationNames);
81  mrs_string onObsNames("");

83  != inObsNames.end(); it++)
84  {
85  for (int lag = 0; lag <= maxLag; lag++)
86  {
87  ostringstream oss;
88  if (normalize && !(doNotNormalizeForLag0 && lag == 0))
89  {
90  oss << "Normalized";
91  }
92  oss << "Autocorr" << lag << "_" << (*it) << ",";
93  onObsNames += oss.str();
94  }

96  }

101 void RunningAutocorrelation::myUpdate(MarControlPtr sender)
102 {
103  (void) sender; //suppress warning of unused parameter(s)
104  MRSDIAG("RunningAutocorrelation.cpp - RunningAutocorrelation:myUpdate");
105 
106  // Update the cache of the control values.
107  maxLag_ = ctrl_maxLag_->to<mrs_natural> ();
108  if (maxLag_ < 0)
109  {
110  MRSERR("maxLag should be greater than zero.");
111  // Setting it to zero to be sure.
112  maxLag_ = 0;
113  }
114  normalize_ = ctrl_normalize_->to<mrs_bool> ();
115  doNotNormalizeForLag0_ = ctrl_doNotNormalizeForLag0_->to<mrs_bool> ();
116  unfoldToObservations_ = ctrl_unfoldToObservations_->to<mrs_bool> ();
117 
118  // Configure output flow.
119  mrs_natural onObservations = unfoldToObservations_ ? inObservations_
120  * (maxLag_ + 1) : inObservations_;
121  mrs_natural onSamples = unfoldToObservations_ ? 1 : maxLag_ + 1;
122  ctrl_onSamples_->setValue(onSamples, NOUPDATE);
123  ctrl_onObservations_->setValue(onObservations, NOUPDATE);
124  ctrl_osrate_->setValue(ctrl_israte_, NOUPDATE);
125  // Handle the observation names.
126  mrs_string onObsNames = ctrl_inObsNames_->to<mrs_string> ();
127  if (unfoldToObservations_)
128  {
129  onObsNames
130  = prefixObservationNamesWithAutocorrelationUnfoldingPrefixes_(
131  onObsNames, normalize_, doNotNormalizeForLag0_, maxLag_);
132  }
133  ctrl_onObsNames_->setValue(onObsNames, NOUPDATE);
134 
135  // Allocate and initialize the buffers and counters.
136  this->acBuffer_.stretch(inObservations_, maxLag_ + 1);
137  this->acBuffer_.setval(0.0);
138  this->memory_.stretch(inObservations_, maxLag_);
139  this->memory_.setval(0.0);
140 
141  // We just cleared the internal buffers, so we should reset the clear control.
142  ctrl_clear_->setValue(false, NOUPDATE);
143 }

145 void RunningAutocorrelation::myProcess(realvec& in, realvec& out)
146 {
148  for (mrs_natural i = 0; i < inObservations_; ++i)
149  {
150  // Calculate the autocorrelation terms
151  for (mrs_natural lag = 0; lag <= maxLag_; lag++)
152  {
153  // For the first part, we need the memory.
154  for (mrs_natural n = 0; n < min(lag, inSamples_); n++)
155  {
156  acBuffer_(i, lag) += in(i, n) * memory_(i, maxLag_ - lag + n);
157  }
158  // For the second part, we have enough with the input slice.
159  for (mrs_natural n = lag; n < inSamples_; n++)
160  {
161  acBuffer_(i, lag) += in(i, n) * in(i, n - lag);
162  }
163  }
164 
165  // Store result in output vector.
166  mrs_natural u = (mrs_natural) unfoldToObservations_;
167  mrs_natural o_base = unfoldToObservations_ ? i * (maxLag_ + 1) : i;
168  for (mrs_natural lag = 0; lag <= maxLag_; lag++)
169  {
170  out(o_base + u * lag, (1 - u) * lag) = acBuffer_(i, lag);
171 
172  }
173  // Do the (optional) normalization.
174  if (normalize_ && acBuffer_(i, 0) > 0.0)
175  {
176  mrs_natural lag = doNotNormalizeForLag0_ ? 1 : 0;
177  for (; lag <= maxLag_; lag++)
178  {
179  out(o_base + u * lag, (1 - u) * lag) /= acBuffer_(i, 0);
180  }
181  }
182 
183  // Remember the last samples for next time.
184  // First, shift samples in memory (if needed).
185  for (mrs_natural m = 0; m < maxLag_ - inSamples_; m++)
186  {
187  memory_(i, m) = memory_(i, m + inSamples_);
188  }
189  // Second, copy over samples from input.
190  for (mrs_natural m = 1; m <= min(maxLag_, inSamples_); m++)
191  {
192  memory_(i, maxLag_ - m) = in(i, inSamples_ - m);
193  }
194  }
195 }
