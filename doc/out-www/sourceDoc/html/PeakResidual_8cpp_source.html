90 PeakResidual::myProcess(realvec& in, realvec& out)
91 {
92  mrs_bool inDb = getControl ("mrs_bool/snrInDb")->to<mrs_bool>();
93  mrs_natural t,o;
94  mrs_real snr = (inDb)? -80.0 : 0.;
95  mrs_real originalPower;
96  mrs_real synthPower;
97  mrs_real diffPower;
98 
99 #ifdef MARSYAS_MATLAB
100 #ifdef MTLB_DBG_LOG
101  MATLAB_PUT(in, "in");
102  MATLAB_EVAL("figure(31),subplot(211),plot(in(1,:)),axis([1 512 -1 1]),grid on,subplot(212),plot(in(2,:)),axis([1 512 -1 1]),grid on");
103 #endif
104 #endif
105 
106  for (o=0; o < inObservations_/2; o++)
107  {
108  originalPower=0;
109  synthPower=0;
110  diffPower=0;
111  for (t = 0; t < inSamples_; t++)
112  {
113  out(o,t) = in(o,t)-in(o+1, t);
114  synthPower += in(o, t)*in(o, t);
115  diffPower += out(o, t)*out(o, t);
116  originalPower += in(o+1, t)*in(o+1, t);
117  }
118 
119  //originalPower/=inSamples_;
120  //synthPower/=inSamples_;
121  //diffPower/=inSamples_;
122 
123  if(synthPower > .001 && originalPower > .01) //[?]
124  {
125  snr = (originalPower)/(diffPower+MINREAL);
126  if (inDb)
127  snr = 10*log10(snr); // +synthPower
128  }
129  }
130 
131  ctrl_SNR_->setValue(snr);
132 
133  if (outFile_.good ())
134  {
135  outFile_ << snr << std::endl;
136  //outFile_.flush ();
137  }
138 }
