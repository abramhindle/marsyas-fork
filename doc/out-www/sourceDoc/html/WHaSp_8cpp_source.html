63 void
64 WHaSp::createSimMatrixNet()
65 {
66  if(HWPSnet_)
67  return;
68 
69  HWPSnet_ = new Series("HWPSnet");
70 
71  //add a feat selector and
72  //set the features needed for HWPS
73  MarSystem* peFeatSelect = new PeakFeatureSelect("peFeatSelect");
74  peFeatSelect->updControl("mrs_natural/selectedFeatures",
75  PeakFeatureSelect::pkFrequency | PeakFeatureSelect::pkSetFrequencies| PeakFeatureSelect::pkSetAmplitudes);
76  HWPSnet_->addMarSystem(peFeatSelect);
77 
78  //create a similarityMatrix MarSystem that uses the HPWS metric
79  SelfSimilarityMatrix* simMat = new SelfSimilarityMatrix("simMat");
80  simMat->addMarSystem(new HWPS("hwps"));
81 
82  HWPSnet_->addMarSystem(simMat);
83 
84  //link totalNumPeaks control to PeakFeatureSelect
85  HWPSnet_->getctrl("PeakFeatureSelect/peFeatSelect/mrs_natural/totalNumPeaks")->linkTo(ctrl_totalNumPeaks_, NOUPDATE);
86  HWPSnet_->update(); //only call update to HWPSnet_ since this is being called from WHaSp::update()! -> avoid potential infinite recursion!
87 
88  //link frameMaxNumPeaks control to PeakFeatureSelect
89  HWPSnet_->getctrl("PeakFeatureSelect/peFeatSelect/mrs_natural/frameMaxNumPeaks")->linkTo(ctrl_frameMaxNumPeaks_, NOUPDATE);
90  HWPSnet_->update(); //only call update to HWPSnet_ since this is being called from WHaSp::update()! -> avoid potential infinite recursion!
91 
92  //link histSize control to HWPS metric
93  HWPSnet_->getctrl("SelfSimilarityMatrix/simMat/HWPS/hwps/mrs_natural/histSize")->linkTo(ctrl_histSize_, NOUPDATE);
94  HWPSnet_->update(); //only call update to HWPSnet_ since this is being called from WHaSp::update()! -> avoid potential infinite recursion!
95 
96  HWPSnet_->setctrl("SelfSimilarityMatrix/simMat/HWPS/hwps/mrs_natural/histSize", 20);
97  HWPSnet_->update(); //only call update to HWPSnet_ since this is being called from WHaSp::update()! -> avoid potential infinite recursion!
98 
99  HWPSnet_->updControl("SelfSimilarityMatrix/simMat/HWPS/hwps/mrs_bool/calcDistance", true);
100 
101  //HWPSnet_->setctrl("SelfSimilarityMatrix/simMat/HWPS/hwps/mrs_natural/histSize", 100);
102  HWPSnet_->update(); //only call update to HWPSnet_ since this is being called from WHaSp::update()! -> avoid potential infinite recursion!
103 
104 }

128  //copy input to output, so we can fill in

132  mrs_natural numPeaks = inPkView.getFrameNumPeaks();
133 
134  if(numPeaks > 0)
135  {
136  simMatrix_.create(numPeaks, numPeaks);
137 
138  HWPSnet_->process(in, simMatrix_);
139 
140  mrs_real maxDist = MINREAL;
141  realvec volumes(numPeaks);
142 
143  for(mrs_natural r=0; r<numPeaks; ++r)
144  {
145  outPkView(r, peakView::pkVolume) = 0;
146  for(mrs_natural c=0; c < numPeaks; ++c)
147  {
148  if(r != c)
149  outPkView(r, peakView::pkVolume) += simMatrix_(r, c)*outPkView(c, peakView::pkAmplitude);
150  }
151  if(outPkView(r, peakView::pkVolume) > maxDist)
152  maxDist = outPkView(r, peakView::pkVolume);
153  volumes(r) = outPkView(r, peakView::pkVolume);
154  }
155 
156  volumes.sort();
157  // cout << "Number of Peaks considered" << numPeaks << endl;

167  if(volumes(nbSelected-j-1) == outPkView(i, peakView::pkVolume))
