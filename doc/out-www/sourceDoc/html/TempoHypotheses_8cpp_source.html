115 void
116 TempoHypotheses::myProcess(realvec& in, realvec& out)
117 {
118  //timeElapsed_ is constantly updated with the referee's next time frame
119  timeElapsed_ = ctrl_tickCount_->to<mrs_natural>();
120 
121  //cout << "THyp: " << timeElapsed_ << "; Ind: " << inductionSize_ << "; accSize: " << accSize_
122  // << "; maxPer: " << maxPeriod_ << "; minPer: " << minPeriod_ << endl;
123 
124  triggerInduction_ = ctrl_triggerInduction_->to<mrs_bool>();
125  if(triggerInduction_)
126  {
127  mrs_natural maxPeriod = 0;
128 
129  //reset flags
130  foundPeriods_ = false;
131  foundPhases_ = false;
132  if(!dumbInductionRequest_) //if not in dumb induction mode (requested by user)
133  {
134  //retrieve Max Period Peak and check if found periods and/or phases
135  mrs_real maxPeriodPeak = 0.0;
136  for (int i=0; i < nPeriods_; i++)
137  {
138  if(in(0, 2*i+1) > 1) //if found any period (period > 1 because it may appear as decimals meaning 0)
139  foundPeriods_ = true;
140 
141  if(in(0, 2*i) > maxPeriodPeak)
142  maxPeriodPeak = in(0, 2*i);
143 
144  //keep maximum period
145  if(in(0, 2*i+1) > maxPeriod)
146  maxPeriod = (mrs_natural)in(0, 2*i+1);
147 
148  int z = 0;
149  //cout << "TH-Phases: " << endl;
150  for (int j = (i * nPhases_); j < ((i+1) * nPhases_); j++)
151  {
152  //cout << "i: " << in(1, 2*z+1) << "; ";
153  if(in(1, 2*z+1) > 0) //if found any phases
154  foundPhases_ = true;
155  //foundPhases_ = false;
156  z++;
157  }
158  //cout << endl;
159  }
160 
161  //cout << "FoundPer: " << foundPeriods_ << "; foundPh: " << foundPhases_ << endl;
162 
163  if(foundPeriods_) //if found periods
164  {
165  for (int i=0; i < nPeriods_; i++)
166  {
167  int z = 0;
168  for (int j = (i * nPhases_); j < ((i+1) * nPhases_); j++)
169  {
170  out(j, 0) = in(0, 2*i+1); //Periods
171  out(j, 1) = in(1, 2*z+1); //Phases
172  out(j, 2) = in(0, 2*i);// / maxPeriodPeak; //Normalized period peak magnitudes
173 
174  z++;
175  }
176  }
177  }
178  }
179 
180  //if no periods found or in dumb induction mode (requested by user)
181  if(!foundPeriods_ || dumbInductionRequest_)
182  {
183  //Manual assorted values for filling BPM hypotheses vector when none are generated
184  int manualBPMs_[] = {120, 60, 240, 100, 160, 200, 80, 140, 180, 220, 150};
185 
186  if(!foundPeriods_ && !dumbInductionRequest_)
187  cerr << "\nUnable to find salient periodicities within the given induction window..." << endl;
188  if(dumbInductionRequest_)
189  cerr << "\nDumb Induction Mode..." << endl;
190 
191  cerr << "...Replacing induction with the following BPMs: ";
192 
193  mrs_natural assignedPerCount = 0;
194  for (unsigned int i=0; i < (sizeof(manualBPMs_) / sizeof(int)); i++)
195  {
196  if(assignedPerCount == nPeriods_) break;
197 
198  mrs_natural manualPeriod = (mrs_natural) (((mrs_real) 60 / (manualBPMs_[i] * hopSize_)) * (srcFs_));
199 
200  //cout << i << "-> manBPM: " << manualBPMs_[i] << "[" << manualPeriod << "] maxPer: "
201  // << maxPeriod_ << "; minPer: " << minPeriod_ << "; ASS: " << assignedPerCount << endl;
202 
203  //assure that the chosen manual periods are within the user-defined BPM range
204  if(manualPeriod >= minPeriod_ && manualPeriod <= maxPeriod_)
205  {
206  cerr << manualBPMs_[i] << "; ";
207 
208  int z = 0;
209  for (int j = (assignedPerCount * nPhases_); j < ((assignedPerCount+1) * nPhases_); j++)
210  {
211  out(j, 0) = manualPeriod; //Periods
212  out(j, 1) = in(1, 2*z+1); //Phases
213  out(j, 2) = 1.0; //equal (max) peak sizes to all manual periods
214  z++;
215 
216  if(out(j,0) > maxPeriod) //save maximum manual period
217  maxPeriod = (mrs_natural) out(j,0);
218  }
219 
220  assignedPerCount++;
221  }
222  }
223 
224  //request dumbInduction to PhaseLock
225  ctrl_dumbInduction_->setValue(true);
226  }
227 
228  if(!foundPhases_) //if no phases found
229  {
230  //cerr << "\nUnable to find potential phases (onsets) within the given induction window..." << endl;
231  //cerr << "...Assuming the maximum nr. of possible phases." << endl;
232 
233  //calculate minimum spacing between possible phases
234  mrs_natural spacing = (mrs_natural) ceil(((mrs_real)maxPeriod / (mrs_real)nPhases_));
235  //start phases on the analysis start point of the given induction window
236  mrs_natural accBeginning = (accSize_-1-inductionSize_);
237 
238  //fill phases vector
239  mrs_realvec phases(nPhases_);
240  mrs_natural index = 0;
241  for(int ph = accBeginning; ph <= accBeginning+maxPeriod+spacing; ph += spacing)
242  {
243  if(index == nPhases_) break;
244 
245  phases(index) = ph; //Phases
246  index++;
247  }
248 
249  for (int i=0; i < nPeriods_; i++)
250  {
251  int z = 0;
252  for (int j = (i * nPhases_); j < ((i+1) * nPhases_); j++)
253  {
254  out(j, 1) = phases(z); //Phases
255  z++;
256  }
257  }
258  //MATLAB_PUT(in, "BeatData");
259  //MATLAB_PUT(out, "TempoHypotheses");
260  }
261 
262  //MATLAB_PUT(out, "TempoHypotheses2");
263  //MATLAB_PUT(in, "BeatData");
264  //MATLAB_EVAL("plot(BeatData)");
265  }
266 
267  /*
268  //FOR TESTING ONSETS IN REPEATED INDUCTION=========================
269  MATLAB_PUT((mrs_natural)triggerInduction_, "Induction_flag");
270  MATLAB_EVAL("Induction = Induction2;");
271  MATLAB_EVAL("Induction(end) = Induction_flag*max(PeakerOnset_in);");
272  */
273 }
