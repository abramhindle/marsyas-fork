52 static void FreqSmear (mrs_realvec &spectrum)
53 {
54  mrs_natural length = spectrum.getSize ();
55  mrs_real buf[3] = {0,0,0},
56  coeff[3] = {.25, .5, .25};
57 
58  spectrum(0) = spectrum(length-1) = 0;
59 
60  for (mrs_natural i = 0; i < length-1;i++)
61  {
62  buf[(i+1)%3] = spectrum(i+1);
63  spectrum(i) = coeff[0]*buf[(i-1+3)%3] + coeff[1]*buf[(i)%3] + coeff[2]*buf[(i+1)%3];
64  }
65 
66  return;
67 }

342 {
343  const unsigned int maxLobeWidth = 6;
344  unsigned int nbP=index.getSize();
345  unsigned int minIndex = 0,
346  endLoop,
347  length = mag.getSize ();
348 
349  // we could also use the instantaneous frequency here...?
350 
351  for(unsigned int i=0 ; i<nbP ; i++)
352  {
353  unsigned int idx = (unsigned int)(index(i)+.1);
354 
355  if (idx <= 0)
356  continue;
357 
358  endLoop = min(length,idx + maxLobeWidth);
359  minIndex = endLoop;
360  // look for the next valley location upward
361  for (unsigned int j = idx ; j < endLoop ; j++)
362  {
363  if(mag(j) < mag(j+1))
364  {
365  minIndex = j;
366  break;
367  }
368  }
369 
370  interval(2*i+1) = minIndex;
371 
372  endLoop = max((unsigned int)0,idx - maxLobeWidth);
373  minIndex = endLoop;
374 
375  // look for the next valley location downward
376  for (unsigned int j= idx ; j > endLoop ; j--)
377  {
378  if(mag(j) < mag(j-1))
379  {
380  minIndex = j;
381  break;
382  }
383  }
384 
385  interval(2*i) = minIndex;

402 void PeakConvert2::ComputeMagnitudeAndPhase (mrs_realvec in)
403 {
404  mrs_real a, c;
405  mrs_real b, d;
406  mrs_real phasediff;
407  for (mrs_natural o=0; o < size_; o++)
408  {
409  if (o==0) //DC bins
410  {
411  a = in(0);
412  b = 0.0;
413  c = in(N_);
414  d = 0.0;
415  }
416  else if (o == size_-1) //Nyquist bins
417  {
418  a = in(1);
419  b = 0.0;
420  c = in(N_+1);
421  d = 0.0;
422  }
423  else //all other bins
424  {
425  a = in(2*o);
426  b = in(2*o+1);
427  c = in(N_+2*o);
428  d = in(N_+2*o+1);
429  }
430 
431  if (o < downFrequency_ || o > upFrequency_)
432  {
433  frequency_(o) = 0;
434  mag_(o) = sqrt((a*a + b*b))*2;
435  continue;
436  }
437  if ( a == .0 || c == .0)
438  {
439  frequency_(o) = o*fundamental_;
440  }
441  else
442  {
443  if(prec_ && pick_)
444  {
445  mrs_real Omega = TWOPI*o*instFreqHopSize_/N_; // now works with hopsizes != 1 as well (AL)
446 
447  // compute phase
448  phase_(o) = atan2(b,a);
449 
450  // compute precise frequency using the phase difference
451  lastphase_(o) = atan2(d,c);
452  phasediff = princArg(phase_(o)-lastphase_(o) - Omega) + Omega;
453  frequency_(o) = abs(phasediff * factor_ );
454  }
455  else
456  frequency_(o) = o*fundamental_;
457  }
458 
459 
460  // compute precise amplitude
461  mag_(o) = sqrt((a*a + b*b))*2;
462  if (pick_)
463  {
464  mrs_real mag = lobe_value_compute((o * fundamental_-frequency_(o))/factor_, 1, N_);
465  magCorr_(o) = mag_(o)/mag;
466  }
467  else
468  {
469  magCorr_(o) = mag_(o);
470  }
471  //mrs_real freq = frequency_(o);
472 
473  if(frequency_(o) != 0.0)
474  {
475  const mrs_natural searchRange = 8;
476  mrs_natural k,
477  kd = o,
478  ku = o,
479  kEndSearch;
480  mrs_real diff[2];
481 
482  // find closest preceding frequency
483  kEndSearch = max ((mrs_natural)0, o-searchRange);
484  for (k = o-1; k > kEndSearch; k--)
485  {
486  diff[0] = abs(frequency_(o) - lastfrequency_(k));
487  diff[1] = abs(frequency_(o) - lastfrequency_(kd));
488  if (diff[0] < diff[1])
489  kd = k;
490  }
491  kEndSearch = min (size_, o+searchRange);
492  for (k = o+1; k < kEndSearch; k++)
493  {
494  diff[0] = abs(frequency_(o) - lastfrequency_(k));
495  diff[1] = abs(frequency_(o) - lastfrequency_(ku));
496  if (diff[0] < diff[1])
497  ku = k;
498  }
499  diff[0] = abs(frequency_(o) - lastfrequency_(kd));
500  diff[1] = abs(frequency_(o) - lastfrequency_(ku));
501  k = (diff[0] < diff[1])? kd : ku;
502 
503  deltafrequency_(o) = (lastfrequency_(k) == 0)? .0 : log10(lastfrequency_(k)/frequency_(o));
504  //deltafrequency_(o) = (frequency_(o)-lastfrequency_(o))/(frequency_(o)+lastfrequency_(o));
505  }
506  else
507  deltafrequency_(o) = .0;
508 
509  mrs_real lastmag = lastmag_(o);
510  if (o > 0)
511  lastmag = max(lastmag_(o-1), lastmag);
512  if (o < size_-1)
513  lastmag = max(lastmag_(o+1), lastmag);
514 
515  if (mag_(o) > 0)
516  deltamag_(o) = (mag_(o)-lastmag)/mag_(o);
517  else if (lastmag > 0)
518  deltamag_(o) = (mag_(o)-lastmag)/lastmag;
519  else
520  deltamag_(o) = 0;
521  }
522  lastfrequency_ = frequency_;
523  lastmag_ = mag_;
524 }

549 void
550 PeakConvert2::myProcess(realvec& in, realvec& out)
551 {
552  mrs_natural o,i;
553  out.setval(0);
554  peakView pkViewOut(out);
555 
556  const mrs_bool useMasking = getctrl("mrs_bool/useMasking")->to<mrs_bool>();
557  const mrs_real probThresh = getctrl("mrs_real/probabilityTresh")->to<mrs_real>();
558 
559  max_->updControl("mrs_natural/nMaximums", frameMaxNumPeaks_);
560 
561  max_->setctrl("mrs_natural/inSamples", size_);
562  max_->setctrl("mrs_natural/inObservations", 1);
563  max_->update();
564  tmp_.stretch(frameMaxNumPeaks_*2);
565 
566  for(mrs_natural f=0 ; f < inSamples_; ++f)
567  {
568  //we should avoid the first empty frames,
569  //that will contain silence and consequently create
570  //discontinuities in the signal, ruining the peak calculation!
571  //only process if we have a full data vector (i.e. no zeros)
572  if(frame_ >= skip_)
573  {
574  // get pair of ffts
575  in.getCol (f, tmpBuff_);
576 
577  // compute magnitude, phase, and instantaneous frequency
578  this->ComputeMagnitudeAndPhase (tmpBuff_);
579 
580  // compute masking threshold
581  if (useMasking && pick_)
582  ComputeMasking (tmpBuff_);
583  else
584  masked_.setval(10.);
585 
586  // select bins with local maxima in magnitude (--> peaks)
587  peaks_ = mag_;
588  if(pick_)
589  this->ComputePeaker (mag_, peaks_);
590  else
591  {
592  for (o = 0 ; o < downFrequency_ ; o++)
593  peaks_(o)=0.0;
594  for (o = upFrequency_ ; o < (mrs_natural)peaks_.getSize() ; o++)
595  peaks_(o)=0.0;
596  }
597 
598  if (lpCoeff_ > 0)
599  FreqSmear (lpPeakerRes_);
600 
601  //compute the probability of a peak being a peak
602  for(o=0 ; o < size_ ; o++)
603  {
604  if (peaks_(o) <= 0)
605  {
606  frequency_(o) = .0;
607  //lastmag_(o) = .0;
608  lastfrequency_(o) = .0;
609  // time smearing if no new peak
610  lpPeakerRes_(o) *=lpCoeff_;
611  continue;
612  }
613 #ifdef ORIGINAL_VERSION
614  // probability of peak being a masker
615  peakProb_(0) = 0;
616  // probability of peak being stationary
617  peakProb_(1) = 0;
618  // probability of peak being tonal
619  peakProb_(2) = (abs(frequency_(o)/fundamental_-o) > .5)? 0 : 1;
620 #else
621  // probability of peak being a masker
622  peakProb_(0) = max(.1, .5 * (log10(masked_(o)) +1.));
623  // probability of peak being stationary
624  peakProb_(1) = max(.1, lpPeakerRes_(o));
625  // probability or peak being tonal
626  peakProb_(2) = GaussianPdf (frequency_(o)/fundamental_-o, gaussianStd);
627 #endif
628 
629  // reset lpPeakerRes with peaker results
630  lpPeakerRes_(o) = 1;
631 
632  peakProb_ *= peakProbWeight_;
633  if ((peakProb_.sum() < probThresh) && pick_)
634  {
635  peaks_(o) = .0;
636  frequency_(o) = .0;
637  //lastmag_(o) = .0;
638  lastfrequency_(o) = .0;
639  }
640  }
641 
642  // keep only the frameMaxNumPeaks_ highest amplitude local maxima
643  tmp_.setval(0.);
644  max_->process(peaks_, tmp_);
645 
646  nbPeaks_=tmp_.getSize()/2;
647  realvec index_(nbPeaks_); //[!] make member to avoid reallocation at each tick!
648  for (i=0 ; i<nbPeaks_ ; i++)
649  index_(i) = tmp_(2*i+1);
650 
651  // search for bins interval
652  realvec interval_(nbPeaks_*2); //[!] make member to avoid reallocation at each tick!
