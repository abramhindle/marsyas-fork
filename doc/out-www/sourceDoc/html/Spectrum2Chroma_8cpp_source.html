87 void
88 Spectrum2Chroma::myUpdate(MarControlPtr sender)
89 {
90  mrs_natural t,o;
91  (void) sender; //suppress warning of unused parameter(s)
92 
93  ctrl_onSamples_->setValue(ctrl_inSamples_, NOUPDATE);
94  ctrl_onObservations_->setValue(ctrl_nbins_, NOUPDATE);
95  ctrl_osrate_->setValue(ctrl_israte_, NOUPDATE);
96 
97  //if (pnbins_ != ctrl_nbins_->to<mrs_natural>())
98  // {
99  pnbins_ = ctrl_nbins_->to<mrs_natural>();
100  ostringstream oss;
101  if(pnbins_ == 12)
102  {
103  for (mrs_natural n=0; n < pnbins_; n++)
104  {
105  oss << "Chroma_" << noteNames_[n] << "_" << ctrl_inObsNames_->to<mrs_string>();
106  }
107  }
108  else
109  {
110  for (mrs_natural n=0; n < pnbins_; n++)
111  {
112  oss << "Chroma_" << n << "_" << ctrl_inObsNames_->to<mrs_string>();
113  }
114  }
115  ctrl_onObsNames_->setValue(oss.str(), NOUPDATE);
116 
117 // }
118 
120  // calculate the Chroma map
121  // based in the fft2chromamx.m MATLAB script by Dan Ellis
122  // http://www.ee.columbia.edu/~dpwe/resources/matlab/chroma-ansyn/#1
124  if(tinObservations_ != inObservations_ ||
125  tonObservations_ != onObservations_ ||
126  tisrate_ != israte_ ||
127  pmiddleAfreq_ != ctrl_middleAfreq_->to<mrs_real>() ||
128  pweightCenterFreq_ != ctrl_weightCenterFreq_->to<mrs_real>() ||
129  pweightStdDev_ != ctrl_weightStdDev_->to<mrs_real>() )
130  {
131 
132  pmiddleAfreq_ = ctrl_middleAfreq_->to<mrs_real>();
133  pweightCenterFreq_ = ctrl_weightCenterFreq_->to<mrs_real>();
134  pweightStdDev_ = ctrl_weightStdDev_->to<mrs_real>();
135 
136  mrs_natural nbins = ctrl_nbins_->to<mrs_natural>();
137  mrs_natural nbins2 = (mrs_natural)floor(nbins/2.0+0.5); //equivalent to round()
138  mrs_natural N2 = inObservations_; // we get N/2+1 spectrum points at the input...
139  mrs_natural N = (N2-1)*2; //fft size
140 
141  //get the original audio sampling rate
142  mrs_real srate = israte_*N;
143 
144  //calculate the frequencies (in octaves) for each spectrum bin
145  //Note: make up value for 0Hz bin -> 1.5 octaves below bin 1
146  //(so chroma is 50% rotated from bin 1, and bin width is broad)
147  realvec fftfreqbins(N2);
148  for(o=1; o < N2; ++o)
149  {
150  fftfreqbins(o) = nbins * hertz2octs(o * 1.0 / N2 * srate, ctrl_middleAfreq_->to<mrs_real>());
151 
152  }
153  if (N2 > 1)
154  fftfreqbins(0) = fftfreqbins(1)-1.5*nbins;
155 
156 
157 
158 
159  //calculate the bin widths
160  realvec binwidthbins(N2);
161  for(o=0; o < N2-1; ++o)
162  {
163  binwidthbins(o) = fftfreqbins(o+1)-fftfreqbins(o);
164  if(binwidthbins(o) < 1.0)
165  binwidthbins(o) = 1.0;
166  }
167  binwidthbins(N2-1) = 1.0;
168 
169 
170 
171  //calculate chroma mapping
172  chromaMap_.create(nbins, N2); //=wts in Dan Ellis MATLAB code
173  realvec D(nbins, N2);
174  for(o = 0; o < nbins; ++o)
175  {
176  for(t = 0; t < N2; ++t)
177  {
178  D(o,t) = fftfreqbins(t) - o;
179 
180  //project into range -nbins/2 .. nbins/2
181  //(add on fixed offset of 10*nbins to ensure all values are positive
182  //for the fmod remainder operation)
183  D(o,t) = fmod(D(o,t) + nbins2 + 10*nbins, nbins) - nbins2;
184 
185  //Gaussian bumps (2*D(o,t) to make them narrower)
186  chromaMap_(o,t) = exp(-0.5 * pow(2.0*D(o,t)/binwidthbins(t), 2.0) );
187  }
188  }
189 
190 
191 
192  //normalize each column by its RMS value
193  mrs_real colRMS;
194  for(t = 0; t < N2; ++t) //iterate over columns
195  {
196  colRMS = 0.0;
197  //get RMS value for column t
198  for(o = 0; o < nbins; ++o) //iterate over rows
199  {
200  colRMS += (chromaMap_(o,t)*chromaMap_(o,t));
201  }
202  //normalize column t
203  if(colRMS != 0.0)
204  {
205  for(o = 0; o < nbins; ++o) //row
206  {
207  chromaMap_(o,t) = chromaMap_(o,t)/sqrt(colRMS);
208  }
209  }
210  }
211 
212 
213  //Maybe apply scaling for fft bins
214  mrs_real ctroct = ctrl_weightCenterFreq_->to<mrs_real>();
215  mrs_real octwidth = ctrl_weightStdDev_->to<mrs_real>();
216 
217  if(octwidth > 0.0)
218  {
219  for(o = 0; o < nbins; ++o)
220  {
221  for(t = 0; t < N2; ++t)
222  {
223  chromaMap_(o,t) = exp(-0.5 * pow((mrs_real) (fftfreqbins(t)/nbins - ctroct)/octwidth , (mrs_real)2.0) );
224  }
225  }
226  }
227 #ifdef MARSYAS_MATLAB
228 // MATLAB_PUT(chromaMap_, "marCM");
229 // MATLAB_EVAL("CM = fft2chromamx("<<N<<","<<nbins<<","<<srate<<","<<pmiddleAfreq_<<","<<pweightCenterFreq_<<","<<pweightStdDev_<<");");
230 #endif //MARSYAS_MATLAB
231  }

234 void
235 Spectrum2Chroma::myProcess(realvec& in, realvec& out)
236 {
237  //input must contain spectral magnitude/power/density/etc
238  //(e.g. output of PowerSpectrum MarSystem)
239 
240  mrs_natural o,t,i;
241  mrs_real chroma_weight;
242 
243  out.setval(0.0);
244 
245 
246  for(o=0; o< onObservations_; ++o)
247  {
248  for(i=0; i< inObservations_; ++i)
249  {
250  chroma_weight = chromaMap_(o,i);
251 
252  for(t=0; t< inSamples_; ++t)
253  {
254  out(o,t) += in(i,t)*chroma_weight;
255  }
256 
257  }
258  }
