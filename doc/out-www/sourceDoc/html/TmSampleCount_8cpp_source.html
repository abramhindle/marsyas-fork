94 void
95 TmSampleCount::updtimer(std::string cname, TmControlValue value)
96 {
97  bool type_error=false;
98  if (cname=="MarSystem/source") {
99  if (value.getType()==tmcv_marsystem) {
100  setSource(value.toMarSystem());
101  }
102  else type_error=true;
103  }
104  else if (cname=="mrs_string/control") {
105  if (value.getType()==tmcv_string) {
106  setSourceCtrl(value.toString());
107  }
108  else type_error=true;
109  }
110  else {
111  MRSWARN("TmSampleCount::updtimer(string,TmControlValue) unsupported control");
112  }
113  if (type_error) {
114  MRSWARN("TmSampleCount::updtimer(string,TmControlValue) wrong type to "+cname);
115  }
116 }
