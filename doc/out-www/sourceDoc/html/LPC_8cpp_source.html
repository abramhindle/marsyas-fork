97 void
98 LPC::autocorrelationWarped(const realvec& in, realvec& r, mrs_real& pitch, mrs_real lambda)
99 {
100  //*Based on the code from: http://www.musicdsp.org/showone.php?id=137
101 
102  //find the order-P autocorrelation array, R, for the sequence x
103  //of length L and using a warping factor of lambda
104 
105  mrs_real* x = in.getData();
106  mrs_natural L = in.getSize();
107  mrs_real* R = r.getData();
108  mrs_natural P = in.getSize()/2;//order_;
109 
110  mrs_real* dl = new mrs_real[L];
111  mrs_real* Rt = new mrs_real[L];
112  mrs_real r1,r2,r1t;
113  R[0]=0;
114  Rt[0]=0;
115  r1=0;
116  r2=0;
117  r1t=0;
118  for(mrs_natural k=0; k<L;k++)
119  {
120  Rt[0]+=x[k]*x[k];
121 
122  dl[k]=r1-lambda*(x[k]-r2);
123  r1 = x[k];
124  r2 = dl[k];
125  }
126  for(mrs_natural i=1; i<=P; ++i)
127  {
128  Rt[i]=0;
129  r1=0;
130  r2=0;
131  for(mrs_natural k=0; k<L;k++)
132  {
133  Rt[i]+=dl[k]*x[k];
134 
135  r1t = dl[k];
136  dl[k]=r1-lambda*(r1t-r2);
137  r1 = r1t;
138  r2 = dl[k];
139  }
140  }
141 
142  for(long i=0; i<=P; ++i)
143  R[i]=Rt[i]/in.getSize(); // [ML] change /
144 
145  delete[] dl;
146  delete[] Rt;
147 
148  //----------------------------------------------------
149  // estimate pitch
150  //----------------------------------------------------
151  mrs_real temp = r(0);
152  //set peak searching start point to 2% of total window size [?]
153  mrs_real j = (mrs_real)in.getSize() * 0.02;
154  //detect first local minimum...
155  while (r((mrs_natural)j) < temp && j < in.getSize()/2)
156  {
157  temp = r((mrs_natural)j);
158  j++;
159  }
160  //... and then from there, detect higher peak => period estimate!
161  temp = 0.0;
162  for (mrs_natural i=(mrs_natural)j; i < in.getSize() * 0.5; ++i)
163  {
164  if (r(i) > temp)
165  {
166  j = i;
167  temp = r(i);
168  }
169  }
170 
171  // This code is from the original Marsyas0.1 AutoCorrelation class
172  //this seems like some sort of Narrowed Autocorrelation (NAC)... [?][!]
173  //however, it does not seem to fit the NAC definition...
174  //so, what is this for??
175  // mrs_real norm = 1.0 / (mrs_real)in.getSize();
176  // mrs_natural k = in.getSize()/2;
177  // for (mrs_natural i=0; i < in.getSize()/2; ++i)
178  // r(i) *= (k-i) * norm;
179 
180  //if autocorr peak not very prominent => not a good period estimate
181  //so discard it...
182  if ((r((mrs_natural)j) / r(0)) < 0.4) j=0;
183  //avoid detection of too high fundamental freqs (e.g. harmonics)?
184  if (j > in.getSize()/4) j = 0;
185 
186  //pitch gets in fact the period (i.e. autocorr lag)
187  //measured in time samples... maybe this should be converted
188  //to a more convenient representation (e.g. pitch in Hz).
189  //Such a conversion would have to depend on the warp factor lambda... [!]
190  pitch = (mrs_real)j;
191 
192  //MATLAB engine stuff - for testing and validation purposes only!
193 #ifdef _MATLAB_LPC_
194  MATLAB_PUT(pitch, "pitch");
195  MATLAB_EVAL("LPC_pitch = [LPC_pitch pitch];");
196 #endif
197 }

200 void
201 LPC::LevinsonDurbin(const realvec& r, realvec& a, realvec& kVec, mrs_real& e)
202 {
203  //*Based on the code from: http://www.musicdsp.org/showone.php?id=137
204 
205  mrs_natural P = order_;
206  mrs_real* R = r.getData();
207  mrs_real* A = a.getData();
208  mrs_real* K = kVec.getData();
209  e = 0.0; //prediction error;
210 
211  mrs_real Am1[62];
212 
213  if(R[0]==0.0)
214  {
215  for(mrs_natural i=1; i<=P; ++i)
216  {
217  K[i]=0.0;
218  A[i]=0.0;
219  }
220  }
221  else
222  {
223  mrs_real km,Em1,Em;
224  mrs_natural k,s,m;
225 
226  for (k=0;k<=P;k++){
227  A[k]=0;
228  Am1[k]=0; }
229 
230  A[0]=1;
231  Am1[0]=1;
232  km=0;
233  Em1=R[0];
234 
235  for (m=1;m<=P;m++) //m=2:N+1
236  {
237  mrs_real err=0.0f; //err = 0;
238 
239  for (k=1;k<=m-1;k++) //for k=2:m-1
240  err += Am1[k]*R[m-k]; // err = err + am1(k)*R(m-k+1);
241 
242  km = (R[m]-err)/Em1; //km=(R(m)-err)/Em1;
243  K[m-1] = -km;
244  A[m]=km; //am(m)=km;
245 
246  for (k=1;k<=m-1;k++) //for k=2:m-1
247  A[k]=Am1[k]-km*Am1[m-k]; // am(k)=am1(k)-km*am1(m-k+1);
248 
249  Em=(1-km*km)*Em1; //Em=(1-km*km)*Em1;
250 
251  for(s=0;s<=P;s++) //for s=1:N+1
252  Am1[s] = A[s]; // am1(s) = am(s)
253 
254  Em1 = Em; //Em1 = Em;
255  //e = Em1; //prediction error
256  e = Em1*Em1; //RMS prediction error
257  }
258  e = sqrt(e/(mrs_real)P); //RMS prediction error
259  }

262 mrs_real
263 LPC::predictionError(const realvec& data, const realvec& coeffs)
264 {
265  mrs_natural i, j;
266  mrs_real power = 0.0;
267  mrs_real error, tmp;
268 
269  //load delay line with input data...
270  for (i=0; i< order_; ++i)
271  {
272  Zs_(i) = data(order_-i-1);
273  }
274  //apply LPC filter and estimate RMS of the error (=~ LPC Gain)
275  mrs_natural count = 0;
276  for (i=order_; i<(mrs_natural)data.getSize() ; ++i)
277  {
278  tmp = 0.0;
279  for (j=0; j< order_; j++)
280  tmp += Zs_(j) * coeffs(j);
281  for (j=order_-1; j>0; j--)
282  Zs_(j) = Zs_(j-1);
283 
284  Zs_(0) = data(i);
285  error = data(i) - tmp;
286  power += error * error;
287  count ++;
288  }
289  return sqrt(power/(mrs_real)count);//=~ RMS LPC Gain
290 }

336 mrs_real
337 LPC::SPcorXpc (mrs_real * rxx, mrs_real * pc, mrs_natural Np)
338 {
339  mrs_natural i, j, k;
340  mrs_real rc, tp;
341  mrs_real perr, t, sum;
342 
343  perr = rxx[0];
344 
345 
346  for (k = 0; k < Np; ++k) {
347 
348  /* Calculate the next reflection coefficient */
349  sum = rxx[k+1];
350  for (i = 0; i < k; ++i)
351  sum = sum - rxx[k-i] * pc[i];
352 
353  /* Check for zero prediction error */
354  if (perr == 0.0 && sum == 0.0)
355  rc = 0.0;
356  else
357  rc = (-sum / perr);
358 
359  /* Calculate the error (equivalent to perr = perr * (1-rc^2))
360  A change in sign of perr means that rc (reflection coefficient for stage k)
361  has a magnitude greater than unity (corresponding to an unstable synthesis
362  filter)
363  */
364  t = perr + rc * sum;
365 
366  perr = t;
367 
368  /* Convert from reflection coefficients to predictor coefficients */
369  pc[k] = -rc;
370  for (i = 0, j = k - 1; i < j; ++i, --j) {
371  tp = pc[i] + rc * pc[j];
372  pc[j] = pc[j] + rc * pc[i];
373  pc[i] = tp;
374  }
375  if (i == j)
376  pc[i] = pc[i] + rc * pc[i];
377  }
378 
379  return perr;
380 }
