64  // ... but add one to the number of output observations.

66 
67  // Should the label go in front or at the back of the observations?
68  labelInFront_ = ctrl_labelInFront_->to<mrs_bool>();
69 
70 
71  // Add the annotation name to onObsNames.
72  mrs_string annotationName = ctrl_annotationName_->to<mrs_string>();
73  mrs_string onObsNames = ctrl_inObsNames_->to<mrs_string>();
74  if (labelInFront_)
75  {

77  }
78  else
79  {
80  onObsNames = onObsNames + mrs_string(",") + annotationName;
81  }

84 }

86 void
87 Annotator::myProcess(realvec& in, realvec& out)
88 {
89  mrs_natural o,t;
90  // Get the label to annotate the feature stream with.
91  const mrs_real& label = ctrl_label_->to<mrs_real>();
92 
93  // Copy the input observations to the output and add the label.
94  for (t = 0; t < inSamples_; t++)
95  {
96  for (o = 0; o < inObservations_; o++)
97  {
98  out((int)(labelInFront_) + o, t) = in(o, t);
99  }
100  out(labelInFront_ ? 0 : onObservations_ - 1, t) = label;
101  }
102 }
