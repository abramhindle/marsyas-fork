97 bool F0Analysis::FindCandidateF0s(const realvec& inPeaks,
98  HarmMap& outHarmSums, F2Fs& outF0ToFks) const
99 {
100  /* For each F0 > FLower, search for harmonically related freqs Fks
101  1. Compute harmonic sum (S) and store (S,F0) in the map outHarmSums
102  2. Store all Fks assigned to F0 in m_F0ToFks */
103  outHarmSums.clear();
104  outF0ToFks.clear();
105 
106  mrs_real theFreqStep = SampleRate_/(2.*inPeaks.getSize());
107  for (mrs_natural i=0; i<inPeaks.getSize(); ++i){
108  mrs_real theF0 = (mrs_real)i*theFreqStep;
109 
110  // F0 > FLower
111  if (inPeaks(i)>0 && theF0 >= LowestF0_){
112 
113  // Compute harmonic sum & energy
114  vector<double> theAssignedFks;
115  mrs_real theSum = 0.0f, theNormFactor = 0.0f;
116  for (mrs_natural j=i+1; j<inPeaks.getSize(); j++){
117  if (inPeaks(j)>0){
118  mrs_real theFk = (mrs_real)j*theFreqStep;
119 
120  /* Check whether Fk is one of the considered harmonics of F0:
121  1. Compute k, the closest integer to Fk/F0
122  2. Check whether |Fk/k-F0| <= tolerance x F0 */
123  int k = ROUND((mrs_real)j/(mrs_real)i);
124  if (k > 1 && k <= NrOfHarmonics_ &&
125  abs(theFk/(mrs_real)k-theF0) <= Tolerance_*theF0){
126 
127  theAssignedFks.push_back(theFk);
128  double tmp = pow(Attenuation_,(double)k);
129  theSum += pow(inPeaks(j),Compression_)*tmp;
130  theNormFactor += tmp;
131  }
132  }
133  }
134 
135  // Add F0
136  if (theAssignedFks.size()>0){
137  outHarmSums[pow(inPeaks(i),Compression_*F0Weight_)
138  *pow(theSum/theNormFactor,1.-F0Weight_)] = theF0;
139  outF0ToFks[theF0] = theAssignedFks;
140  }
141  }
142  }
143  return true;
144 }

146 bool F0Analysis::SelectUnrelatedF0s(const realvec& inPeaks,
147  const HarmMap inHarmSums, const F2Fs& inF0ToFks, realvec& outNoteEvidence)
148 {
149  outNoteEvidence.setval(0);
150 
151  FreqMap thePeaks;
152  mrs_real theFreqStep = SampleRate_/(2.*inPeaks.getSize());
153  for (mrs_natural i=0; i<inPeaks.getSize(); ++i)
154  if (inPeaks(i)>0)
155  thePeaks[(mrs_real)i*theFreqStep] = inPeaks(i);
156 
157  ChordEvidence_ = 0.0f;
158  mrs_natural theNrOfPitches = 0;
159  if (!inHarmSums.empty()){
160  /* Select candidates F0 (called Fc) that are not harmonically related.
161  At first, select Fc with the highest harmonic sum (S) value.
162  At second, select the other Fcs in decreasing order of S while
163  Fc is not related to a selected Fc (HARMONIC or SUBHARMONIC)*/
164 
165  HarmMap::const_iterator Cand; // Candidate F0
166  FreqMap::iterator Sel; // Selected F0
167 
168  // Add first candidate
169  Cand = inHarmSums.begin();
170  double F0c = Cand->second;
171  outNoteEvidence(ROUND(F0c/theFreqStep)) =
172  ComputePowerOfF0(thePeaks, inF0ToFks, F0c);
173  theNrOfPitches++;
174  Cand++;
175 
176  // Proceed with other candidates
177  mrs_real theHypPower = ComputePowerOfF0(thePeaks, inF0ToFks, F0c);
178  mrs_real theAllPower = ComputePowerOfInput(thePeaks);
179  while(Cand!=inHarmSums.end()){
180 
181  // Check relationship with selected F0
182  F0c = Cand->second;
183  bool theRelFlag = false;
184 
185  for (mrs_natural i=0; i<outNoteEvidence.getSize(); ++i)
186  {
187  if (outNoteEvidence(i) > 0)
188  {
189  double F0 = (mrs_real)i*theFreqStep;
190  int k = ROUND(F0c/F0);
191  int l = ROUND(F0/F0c);
192 
193  // theRelFlag = true if Sel and Cand are harmonically related
194  theRelFlag = (k > 0 && k <= NrOfHarmonics_ &&
195  abs(F0c/(double)k-F0) <= Tolerance_*F0) ||
196  (l > 0 && l <= NrOfHarmonics_ &&
197  abs((double)l*F0c-F0) <= Tolerance_*F0);
198 
199  if (theRelFlag)
200  break;
201  }
202  }
203 
204  if (!theRelFlag)
205  {
206  outNoteEvidence(ROUND(F0c/theFreqStep)) = ComputePowerOfF0(thePeaks, inF0ToFks, F0c);
207  theHypPower = ComputePowerOfHyp(thePeaks, inF0ToFks, outNoteEvidence);
208  theNrOfPitches++;
209  }
210  Cand++;
211  }
212 
213  // Normalize note relevances
214  mrs_real theFactor = 0.0f;
215  for (mrs_natural i=0; i<outNoteEvidence.getSize(); ++i)
216  theFactor += outNoteEvidence(i);
217  for (mrs_natural i=0; i<outNoteEvidence.getSize(); ++i)
218  outNoteEvidence(i) /= theFactor;
219 
220  // Compute chord evidence if nr of notes >= 2
221  if (theNrOfPitches>=2)
222  ChordEvidence_= theHypPower/theAllPower;
223  }
224  return true;
225 }

255  for (mrs_natural i=0; i<inNoteEvidence.getSize(); ++i)
