130 bool
131 AimGammatone::InitializeInternal() {
132  mrs_natural num_channels = ctrl_num_channels_->to<mrs_natural>();
133  double min_frequency = ctrl_min_frequency_->to<mrs_real>();
134  double max_frequency = ctrl_max_frequency_->to<mrs_real>();
135 
136  // Calculate number of channels, and centre frequencies
137  double erb_max = ERBTools::Freq2ERB(max_frequency);
138  double erb_min = ERBTools::Freq2ERB(min_frequency);
139  double delta_erb = (erb_max - erb_min) / (num_channels - 1);
140 
141  centre_frequencies_.resize(num_channels);
142  double erb_current = erb_min;
143 
144  for (int i = 0; i < num_channels; ++i) {
145  centre_frequencies_[i] = ERBTools::ERB2Freq(erb_current);
146  erb_current += delta_erb;
147  }
148 
149  a_.resize(num_channels);
150  b1_.resize(num_channels);
151  b2_.resize(num_channels);
152  b3_.resize(num_channels);
153  b4_.resize(num_channels);
154  state_1_.resize(num_channels);
155  state_2_.resize(num_channels);
156  state_3_.resize(num_channels);
157  state_4_.resize(num_channels);
158 
159  for (int ch = 0; ch < num_channels; ++ch) {
160  double cf = centre_frequencies_[ch];
161  double erb = ERBTools::Freq2ERBw(cf);
162 
163  // Sample interval
164  double dt = 1.0 / ctrl_israte_->to<mrs_real>();
165 
166  // Bandwidth parameter
167  double b = 1.019 * 2.0 * PI * erb;
168 
169  // The following expressions are derived in Apple TR #35, "An
170  // Efficient Implementation of the Patterson-Holdsworth Cochlear
171  // Filter Bank" and used in Malcolm Slaney's auditory toolbox, where he
172  // defines this alternaltive four stage cascade of second-order filters.
173 
174  // Calculate the gain:
175  double cpt = cf * PI * dt;
176  complex<double> exponent(0.0, 2.0 * cpt);
177  complex<double> ec = exp(2.0 * exponent);
178  complex<double> two_cf_pi_t(2.0 * cpt, 0.0);
179  complex<double> two_pow(pow(2.0, (3.0 / 2.0)), 0.0);
180  complex<double> p1 = -2.0 * ec * dt;
181  complex<double> p2 = 2.0 * exp(-(b * dt) + exponent) * dt;
182  complex<double> b_dt(b * dt, 0.0);
183 
184  double gain = abs(
185  (p1 + p2 * (cos(two_cf_pi_t) - sqrt(3.0 - two_pow) * sin(two_cf_pi_t)))
186  * (p1 + p2 * (cos(two_cf_pi_t) + sqrt(3.0 - two_pow) * sin(two_cf_pi_t)))
187  * (p1 + p2 * (cos(two_cf_pi_t) - sqrt(3.0 + two_pow) * sin(two_cf_pi_t)))
188  * (p1 + p2 * (cos(two_cf_pi_t) + sqrt(3.0 + two_pow) * sin(two_cf_pi_t)))
189  / pow((-2.0 / exp(2.0 * b_dt) - 2.0 * ec + 2.0 * (1.0 + ec)
190  / exp(b_dt)), 4));
191 
192  // The filter coefficients themselves:
193  const int coeff_count = 3;
194  a_[ch].resize(coeff_count, 0.0);
195  b1_[ch].resize(coeff_count, 0.0);
196  b2_[ch].resize(coeff_count, 0.0);
197  b3_[ch].resize(coeff_count, 0.0);
198  b4_[ch].resize(coeff_count, 0.0);
199  state_1_[ch].resize(coeff_count, 0.0);
200  state_2_[ch].resize(coeff_count, 0.0);
201  state_3_[ch].resize(coeff_count, 0.0);
202  state_4_[ch].resize(coeff_count, 0.0);
203 
204  double B0 = dt;
205  double B2 = 0.0;
206 
207  double B11 = -(2.0 * dt * cos(2.0 * cf * PI * dt) / exp(b * dt)
208  + 2.0 * sqrt(3 + pow(2.0, 1.5)) * dt
209  * sin(2.0 * cf * PI * dt) / exp(b * dt)) / 2.0;
210  double B12 = -(2.0 * dt * cos(2.0 * cf * PI * dt) / exp(b * dt)
211  - 2.0 * sqrt(3 + pow(2.0, 1.5)) * dt
212  * sin(2.0 * cf * PI * dt) / exp(b * dt)) / 2.0;
213  double B13 = -(2.0 * dt * cos(2.0 * cf * PI * dt) / exp(b * dt)
214  + 2.0 * sqrt(3 - pow(2.0, 1.5)) * dt
215  * sin(2.0 * cf * PI * dt) / exp(b * dt)) / 2.0;
216  double B14 = -(2.0 * dt * cos(2.0 * cf * PI * dt) / exp(b * dt)
217  - 2.0 * sqrt(3 - pow(2.0, 1.5)) * dt
218  * sin(2.0 * cf * PI * dt) / exp(b * dt)) / 2.0;
219 
220  a_[ch][0] = 1.0;
221  a_[ch][1] = -2.0 * cos(2.0 * cf * PI * dt) / exp(b * dt);
222  a_[ch][2] = exp(-2.0 * b * dt);
223  b1_[ch][0] = B0 / gain;
224  b1_[ch][1] = B11 / gain;
225  b1_[ch][2] = B2 / gain;
226  b2_[ch][0] = B0;
227  b2_[ch][1] = B12;
228  b2_[ch][2] = B2;
229  b3_[ch][0] = B0;
230  b3_[ch][1] = B13;
231  b3_[ch][2] = B2;
232  b4_[ch][0] = B0;
233  b4_[ch][1] = B14;
234  b4_[ch][2] = B2;
235  }
236  return true;
237 }

255 void
256 AimGammatone::myProcess(realvec& in, realvec& out)
257 {
258  int audio_channel = 0;
259 
260  std::vector<std::vector<double> >::iterator b1 = b1_.begin();
261  std::vector<std::vector<double> >::iterator b2 = b2_.begin();
262  std::vector<std::vector<double> >::iterator b3 = b3_.begin();
263  std::vector<std::vector<double> >::iterator b4 = b4_.begin();
264  std::vector<std::vector<double> >::iterator a = a_.begin();
265  std::vector<std::vector<double> >::iterator s1 = state_1_.begin();
266  std::vector<std::vector<double> >::iterator s2 = state_2_.begin();
267  std::vector<std::vector<double> >::iterator s3 = state_3_.begin();
268  std::vector<std::vector<double> >::iterator s4 = state_4_.begin();
269 
270  // Temporary storage between filter stages
271  std::vector<double> outbuff(ctrl_inSamples_->to<mrs_natural>());
272  mrs_natural _channel_count = ctrl_num_channels_->to<mrs_natural>();
273  mrs_natural _inSamples = ctrl_inSamples_->to<mrs_natural>();
274 
275  for (int ch = 0; ch < _channel_count;
276  ++ch, ++b1, ++b2, ++b3, ++b4, ++a, ++s1, ++s2, ++s3, ++s4) {
277  for (int i = 0; i < _inSamples; ++i) {
278  // Direct-form-II IIR filter
279  double inputsample = in(audio_channel, i);
280  outbuff[i] = (*b1)[0] * inputsample + (*s1)[0];
281  for (unsigned int stage = 1; stage < s1->size(); ++stage)
282  (*s1)[stage - 1] = (*b1)[stage] * inputsample
283  - (*a)[stage] * outbuff[i] + (*s1)[stage];
284  }
285  for (int i = 0; i < _inSamples; ++i) {
286  // Direct-form-II IIR filter
287  double inputsample = outbuff[i];
288  outbuff[i] = (*b2)[0] * inputsample + (*s2)[0];
289  for (unsigned int stage = 1; stage < s2->size(); ++stage)
290  (*s2)[stage - 1] = (*b2)[stage] * inputsample
291  - (*a)[stage] * outbuff[i] + (*s2)[stage];
292  }
293  for (int i = 0; i < _inSamples; ++i) {
294  // Direct-form-II IIR filter
295  double inputsample = outbuff[i];
296  outbuff[i] = (*b3)[0] * inputsample + (*s3)[0];
297  for (unsigned int stage = 1; stage < s3->size(); ++stage)
298  (*s3)[stage - 1] = (*b3)[stage] * inputsample
299  - (*a)[stage] * outbuff[i] + (*s3)[stage];
300  }
301  for (int i = 0; i < _inSamples; ++i) {
302  // Direct-form-II IIR filter
303  double inputsample = outbuff[i];
304  outbuff[i] = (*b4)[0] * inputsample + (*s4)[0];
305  for (unsigned int stage = 1; stage < s4->size(); ++stage)
306  (*s4)[stage - 1] = (*b4)[stage] * inputsample
307  - (*a)[stage] * outbuff[i] + (*s4)[stage];
308  out(ch, i) = outbuff[i];
309  }
310  }
