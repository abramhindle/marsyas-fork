79 void
80 NoiseGate::myProcess(realvec& in, realvec& out)
81 {
82  mrs_natural t,o;
83  //checkFlow(in,out);
84 
85  mrs_real thresh = getctrl("mrs_real/thresh")->to<mrs_real>();
86  mrs_real release = getctrl("mrs_real/release")->to<mrs_real>();
87  mrs_real rolloff = getctrl("mrs_real/rolloff")->to<mrs_real>();
88  mrs_real at = getctrl("mrs_real/at")->to<mrs_real>();
89  mrs_real rt = getctrl("mrs_real/rt")->to<mrs_real>();
90  // FIXME This variable is defined but unused.
91  // mrs_real slope = getctrl("mrs_real/slope")->to<mrs_real>();
92 
93  // calculate rolloff, at and rt time
94  at = 1 - exp(-2.2/(22050*at));
95  rt = 1 - exp(-2.2/(22050*rt));
96 
97  for (o = 0; o < inObservations_; o++)
98  for (t = 0; t < inSamples_; t++)
99  {
100  // Calculates the current amplitude of signal and incorporates
101  // the at and rt times into xd(o,t)
102  alpha_ = fabs(in(o,t)) - xdprev_;
103  if (alpha_ < 0)
104  {
105  alpha_ = 0;
106  }
107  xdprev_=xdprev_*(1-rt)+at*alpha_;
108 
109  if (state_ == 1.0)
110  {
111  if (xdprev_ < thresh)
112  {
113  gains_(o,t) = gainsprev_*rolloff;
114  state_ = 0.0;
115  }
116  else
117  {
118  gains_(o,t) = 1;
119  }
120  }
121  else
122  {
123  if (xdprev_ < release)
124  {
125  gains_(o,t) = gainsprev_*rolloff;
126  // rolloff time
127  }
128  else
129  if (xdprev_ > release)
130  {
131  gains_(o,t) = 1.0;
132  state_ = 1.0;
133  }
134  else
135  {
136  gains_(o,t) = 0.0;
137  }
138  }
139 
140  gainsprev_ = gains_(o,t);
141  out(o,t) = gainsprev_ * in(o,t);
142 
143  }
144 }
