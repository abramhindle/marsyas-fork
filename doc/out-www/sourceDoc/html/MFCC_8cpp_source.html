67 void
68 MFCC::myUpdate(MarControlPtr sender)
69 {
70  (void) sender; //suppress warning of unused parameter(s)
71 
72  // Get the number of cepstral coefficients from the control
73  cepstralCoefs_ = ctrl_cepstralCoefs_->to<mrs_natural>();
74 
75  ctrl_onSamples_->setValue((mrs_natural)1, NOUPDATE);
76  ctrl_onObservations_->setValue((mrs_natural)cepstralCoefs_, NOUPDATE);
77  ctrl_osrate_->setValue(ctrl_israte_, NOUPDATE);
78 
79  // Initialize frequency boundaries for filters
80  mrs_natural i,j;
81  fftSize_ = 2 * (ctrl_inObservations_->to<mrs_natural>()-1); //PowerSpectrum outputs N/2+1 "magnitude" spectral points!
82  if (fftSize_ == 0) return; // skip unnecessary updates
83 
84  samplingRate_ = (mrs_natural) (ctrl_israte_->to<mrs_real>() * fftSize_);
85 
86  // Set the observation names: use the first item of the
87  // input observation names and prefix it with "MFCCxx"
88  mrs_string inObsName = stringSplit(ctrl_inObsNames_->to<mrs_string>(), ",")[0];
89  ostringstream oss;
90  for (i=0; i < cepstralCoefs_; ++i)
91  {
92  oss << "MFCC" << i << "_" << inObsName << ",";
93  }
94  ctrl_onObsNames_->setValue(oss.str(), NOUPDATE);
95 
96  if ((pfftSize_ != fftSize_) || (psamplingRate_ != samplingRate_) || (pcepstralCoefs_ != cepstralCoefs_))
97  {
98 
99  freqs_.create(42);
100  lowestFrequency_ = 133.3333f;
101  linearFilters_ = 13;
102  linearSpacing_ = 66.66666f;
103  logFilters_ = 27;
104  logSpacing_ = 1.0711703f;
105 
106  totalFilters_ = linearFilters_ + logFilters_;
107  lower_.create(totalFilters_);
108  center_.create(totalFilters_);
109  upper_.create(totalFilters_);
110  triangle_heights_.create(totalFilters_);
111 
112  // Linear filter boundaries
113  for (i=0; i< linearFilters_; ++i)
114  {
115  freqs_(i) = lowestFrequency_ + i * linearSpacing_;
116  }
117 
118  // Logarithmic filter boundaries
119  mrs_real first_log = freqs_(linearFilters_-1);
120  for (i=1; i<=logFilters_+2; ++i)
121  {
122  freqs_(linearFilters_-1+i) = first_log * pow(logSpacing_, (mrs_real)i);
123  }
124 
125  // Triangles information
126  for (i=0; i<totalFilters_; ++i)
127  {
128  lower_(i) = freqs_(i);
129  center_(i) = freqs_(i+1);
130  upper_(i) = freqs_(i+2);
131  triangle_heights_(i) = (mrs_real)(2.0 / (upper_(i) - lower_(i)));
132  }
133 
134  fftFreqs_.stretch(fftSize_);
135 
136  for (i=0; i< fftSize_; ++i)
137  {
138  fftFreqs_(i) = (float)i / (float)fftSize_ * (float)samplingRate_;
139  }
140 
141  mfccFilterWeights_.create(totalFilters_, fftSize_);
142  mfccDCT_.create(cepstralCoefs_, totalFilters_);
143 
144  mrs_natural chan;
145 
146  // NEIL's filter weight speedup
147  if (pfftSize_!=fftSize_)
148  {
149  if (mfcc_offsets_!=NULL)
150  {
151  delete [] mfcc_offsets_;
152  }
153  mfcc_offsets_ = new int[totalFilters_*fftSize_*2];
154  }
155  // Initialize mfccFilterWeights
156  for (chan = 0; chan < totalFilters_; chan++)
157  {
158  // NEIL's filter weight speedup
159  int len=-1;
160  int pos=0;
161  for (i=0; i< fftSize_; ++i)
162  {
163  if ((fftFreqs_(i) > lower_(chan))&& (fftFreqs_(i) <= center_(chan)))
164  {
165  mfccFilterWeights_(chan, i) = triangle_heights_(chan) *
166  ((fftFreqs_(i) - lower_(chan))/(center_(chan) - lower_(chan)));
167  // NEIL's filter weight speedup
168  if (len==-1)
169  {
170  pos=i;
171  }
172  len=i;
173  }
174  if ((fftFreqs_(i) > center_(chan)) && (fftFreqs_(i) <= upper_(chan)))
175  {
176  mfccFilterWeights_(chan, i) = triangle_heights_(chan) *
177  ((upper_(chan) - fftFreqs_(i))/(upper_(chan) - center_(chan)));
178  // NEIL's filter weight speedup
179  if (len==-1)
180  {
181  pos=i;
182  }
183  len=i;
184  }
185  }
186  // NEIL's filter weight speedup
187  mfcc_offsets_[chan] = pos;
188  mfcc_offsets_[chan+totalFilters_] = len;
189  }
190 
191  // Initialize MFCC_DCT
192  mrs_real scale_fac = (mrs_real)(1.0/ sqrt((mrs_real)(totalFilters_/2)));
193  for (j = 0; j<cepstralCoefs_; j++)
194  {
195  for (i=0; i< totalFilters_; ++i)
196  {
197  mfccDCT_(j, i) = scale_fac * cos(j * (2*i +1) * PI/2/totalFilters_);
198  if (j == 0)
199  {
200  mfccDCT_(j,i) *= (mrs_real)(sqrt(2.0)/2.0);
201  }
202  }
203  }
204  }
205 
206 
207  pfftSize_ = fftSize_;
208  psamplingRate_ = samplingRate_;
209 
210  fmagnitude_.stretch(ctrl_inObservations_->to<mrs_natural>() * 2);
211  earMagnitude_.stretch(totalFilters_);
212 }

214 void
215 MFCC::myProcess(realvec& in, realvec& out)
216 {
217  mrs_natural i,k,o;
218 
219  // mirror the spectrum
220  for (o=0; o < inObservations_; o++)
221  {
222  fmagnitude_(o) = in(o,0);
223  }
224 
225  for (o=0; o< inObservations_; o++)
226  {
227  fmagnitude_(o + inObservations_) = fmagnitude_(inObservations_ - o -1);
228  }
229 
230  mrs_real sum =0.0;
231  // Calculate the filterbank responce
232  for (i=0; i<totalFilters_; ++i)
233  {
234  sum = 0.0;
235  // NEIL's filter weight speedup
236  for (k=mfcc_offsets_[i]; k<=mfcc_offsets_[i+totalFilters_]; k++)
237  {
238  sum += (mfccFilterWeights_(i, k) * fmagnitude_(k));
239  }
240  if (sum != 0.0)
241  {
242  earMagnitude_(i) = log10(sum);
243  }
244  else
245  {
246  earMagnitude_(i) = 0.0;
247  }
248  }
249  /* The way it used to be : NEIL
250  for (i=0; i<totalFilters_; ++i) {
251  sum = 0.0;
252  for (k=0; k<fftSize_; k++) {
253  sum += (mfccFilterWeights_(i, k) * fmagnitude_(k));
254  }
255  if (sum != 0.0)
256  earMagnitude_(i) = log10(sum);
257  else
258  earMagnitude_(i) = 0.0;
259  }
260  */
261 
262  // Take the DCT
263  for (o=0; o < cepstralCoefs_; o++)
264  {
265  sum =0.0;
266  for (k=0; k < totalFilters_; k++)
267  {
268  sum += (mfccDCT_(o,k) * earMagnitude_(k));
269  }
270  out(o,0) = sum;
271  }
272 }
