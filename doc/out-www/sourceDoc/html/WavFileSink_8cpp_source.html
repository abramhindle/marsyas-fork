100 void
101 WavFileSink::putHeader(mrs_string filename)
102 {
103  mrs_natural nChannels = (mrs_natural)getctrl("mrs_natural/inObservations")->to<mrs_natural>();
104  sfp_ = fopen(filename.c_str(), "wb");
105 
106  written_ = 0;
107 
108  /* prepare header */
109  hdr_.riff[0] = 'R';
110  hdr_.riff[1] = 'I';
111  hdr_.riff[2] = 'F';
112  hdr_.riff[3] = 'F';
113 
114  hdr_.file_size = 44;
115 
116  hdr_.wave[0] = 'W';
117  hdr_.wave[1] = 'A';
118  hdr_.wave[2] = 'V';
119  hdr_.wave[3] = 'E';
120 
121  hdr_.fmt[0] = 'f';
122  hdr_.fmt[1] = 'm';
123  hdr_.fmt[2] = 't';
124  hdr_.fmt[3] = ' ';
125 
126 #if defined(MARSYAS_BIGENDIAN)
127  hdr_.chunk_size = ByteSwapLong(16);
128  hdr_.format_tag = ByteSwapShort(1);
129  hdr_.num_chans = ByteSwapShort((signed short)nChannels);
130  hdr_.sample_rate = ByteSwapLong((mrs_natural)getctrl("mrs_real/israte")->to<mrs_real>());
131  hdr_.bytes_per_sec = ByteSwapLong(hdr_.sample_rate * 2);
132  hdr_.bytes_per_samp = ByteSwapShort(2);
133  hdr_.bits_per_samp = ByteSwapShort(16);
134  hdr_.data_length = ByteSwapLong(0);
135 #else
136  hdr_.chunk_size = 16;
137  hdr_.format_tag = 1;
138  hdr_.num_chans = (signed short)nChannels;
139  hdr_.sample_rate = (mrs_natural)getctrl("mrs_real/israte")->to<mrs_real>();
140  hdr_.bytes_per_sec = hdr_.sample_rate * 2;
141  hdr_.bytes_per_samp = 2;
142  hdr_.bits_per_samp = 16;
143  hdr_.data_length = 0;
144 #endif
145 
146  hdr_.data[0] = 'd';
147  hdr_.data[1] = 'a';
148  hdr_.data[2] = 't';
149  hdr_.data[3] = 'a';
150 
151  fwrite(&hdr_, 4, 11, sfp_);
152 
153  sfp_begin_ = ftell(sfp_);
154 }

159  return (((nLongNumber&0x000000FF)<<24)+((nLongNumber&0x0000FF00)<<8)+

170 void
171 WavFileSink::putLinear16Swap(realvec& slice)
172 {
173  for (mrs_natural c=0; c < nChannels_; ++c)
174  for (mrs_natural t=0; t < inSamples_; t++)
175  {
176  #if defined(MARSYAS_BIGENDIAN)
177  sdata_[t*nChannels_ + c] = ByteSwapShort((short)(slice(c,t) * PCM_MAXSHRT));
178  #else
179  sdata_[t*nChannels_ + c] = (short)(slice(c,t) * PCM_MAXSHRT);
180  #endif
181  }
182 
183  if ((mrs_natural)fwrite(sdata_, sizeof(short), nChannels_ * inSamples_, sfp_) != nChannels_ * inSamples_)
184  {
185  MRSERR("Problem: could not write window to file " + filename_);
186  }
187 }

189 void
190 WavFileSink::myProcess(realvec& in, realvec& out)
191 {
192  mrs_natural o,t;
193 
194  // copy input to output
195  for (o=0; o < inObservations_; o++)
196  for (t=0; t < inSamples_; t++)
197  {
198  out(o,t) = in(o,t);
199  }
200 
201  long fileSize;
202  fpos_ = ftell(sfp_);
203 
204  // jump to start and write data size
205  fseek(sfp_, 40, SEEK_SET);
206  written_ += inSamples_;
207  fileSize = (written_ * 2 * nChannels_);
208  #if defined(MARSYAS_BIGENDIAN)
209  fileSize = ByteSwapLong(fileSize);
210  #endif
211 
212  fwrite(&fileSize, 4, 1, sfp_);
213  fseek(sfp_, fpos_, SEEK_SET);
214 
215  putLinear16Swap(in);
216 }
