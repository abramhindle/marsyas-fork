82 void
83 AutoCorrelationFFT::myProcess(realvec& in, realvec& out)
84 {
85  for (mrs_natural o=0;o<inObservations_;o++)
86  {
87 
88  mrs_natural i;
89  temp_.create(inSamples_);
90  for (i=0; i < inSamples_; ++i)
91  temp_(i) = in(o,i);
92  mrs_real *temp = temp_.getData();
93  fft_.rfft(temp, inSamples_/2, FFT_FORWARD);
94  //temp[0] = 0.0;
95  //temp[1] = 0.0;
96  out(o,0) = 0.0; // DC
97 
98 
99  // compute magnitude of freqs
100  for (i=1; i<onSamples_; ++i)
101  {
102  out(o,i) = sqrt(temp[2*i]*temp[2*i] + temp[2*i+1]*temp[2*i+1]);
103  //cout << "outvec[" << 2*i << "]: " << outvec(2*i) << endl;
104  }
105  }
