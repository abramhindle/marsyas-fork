113 // sness - Because we can't pass the centre_frequencies from one
114 // MarSystem to the next, we need to recalculate them here.
115 void
116 AimSSI::CalculateCentreFrequencies() {
117  int num_channels = ctrl_inObservations_->to<mrs_natural>();
118  double erb_max = ERBTools::Freq2ERB(ctrl_max_frequency_->to<mrs_real>());
119  double erb_min = ERBTools::Freq2ERB(ctrl_min_frequency_->to<mrs_real>());
120  double delta_erb = (erb_max - erb_min) / (num_channels - 1);
121 
122  centre_frequencies_.resize(num_channels);
123  double erb_current = erb_min;
124 
125  for (int i = 0; i < num_channels; ++i) {
126  centre_frequencies_[i] = ERBTools::ERB2Freq(erb_current);
127  erb_current += delta_erb;
128  }
129 }

131 int
132 AimSSI::ExtractPitchIndex(realvec& in) const {
133  // Generate temporal profile of the SAI
134  std::vector<double> sai_temporal_profile(ctrl_inSamples_->to<mrs_natural>(), 0.0);
135  for (int i = 0; i < ctrl_inSamples_->to<mrs_natural>(); ++i) {
136  double val = 0.0;
137  for (int ch = 0; ch < ctrl_inObservations_->to<mrs_natural>(); ++ch) {
138  val += in(ch, i);
139  }
140  sai_temporal_profile[i] = val;
141  }
142 
143  // Find pitch value
144  int start_sample = (int)floor(ctrl_pitch_search_start_ms_->to<mrs_real>() * ctrl_israte_->to<mrs_real>() / 1000.0);
145  int max_idx = 0;
146  double max_val = 0.0;
147  for (int i = start_sample; i < ctrl_inSamples_->to<mrs_natural>(); ++i) {
148  if (sai_temporal_profile[i] > max_val) {
149  max_idx = i;
150  max_val = sai_temporal_profile[i];
151  }
152  }
153  return max_idx;
154 }

157 void
158 AimSSI::myProcess(realvec& in, realvec& out)
159 {
160  int pitch_index = ctrl_inSamples_->to<mrs_natural>() - 1;
161  if (ctrl_do_pitch_cutoff_->to<mrs_bool>()) {
162  pitch_index = ExtractPitchIndex(in);
163  }
164 
165  for (o = 0; o < ctrl_inObservations_->to<mrs_natural>(); o++) {
166  double centre_frequency = centre_frequencies_[o];
167  // Copy the buffer from input to output, addressing by h-value
168  for (t = 0; t < ssi_width_samples_; t++) {
169  double h;
170  double cycle_samples = ctrl_israte_->to<mrs_real>() / centre_frequency;
171  if (ctrl_log_cycles_axis_->to<mrs_bool>()) {
172  double gamma_min = -1.0;
173  double gamma_max = Log2(ctrl_ssi_width_cycles_->to<mrs_real>());
174  double gamma = gamma_min + (gamma_max - gamma_min)
175  * static_cast<double>(t)
176  / static_cast<double>(ssi_width_samples_);
177  h = pow(2.0, gamma);
178  } else {
179  h = static_cast<double>(t) * ctrl_ssi_width_cycles_->to<mrs_real>()
180  / static_cast<double>(ssi_width_samples_);
181  }
182 
183  // The index into the input array is a doubleing-point number, which is
184  // split into a whole part and a fractional part. The whole part and
185  // fractional part are found, and are used to linearly interpolate
186  // between input samples to yield an output sample.
187  double whole_part;
188  double frac_part = modf(h * cycle_samples, &whole_part);
189  int sample = (int)floor(whole_part);
190 
191  double weight = 1.0;
192 
193  int cutoff_index = ctrl_inSamples_->to<mrs_natural>() - 1;
194  if (ctrl_do_pitch_cutoff_->to<mrs_bool>()) {
195  if (pitch_index < cutoff_index) {
196  if (ctrl_weight_by_cutoff_->to<mrs_bool>()) {
197  weight *= static_cast<double>(ctrl_inSamples_->to<mrs_natural>())
198  / static_cast<double>(pitch_index);
199  }
200  cutoff_index = pitch_index;
201  }
202  }
203 
204  if (ctrl_weight_by_scaling_->to<mrs_bool>()) {
205  if (centre_frequency > ctrl_pivot_cf_->to<mrs_real>()) {
206  weight *= (centre_frequency / ctrl_pivot_cf_->to<mrs_real>());
207  }
208  }
209 
210  double val;
211  if (sample < cutoff_index) {
212  double curr_sample = in(o, sample);
213  double next_sample = in(o, sample + 1);
214  val = weight * (curr_sample
215  + frac_part * (next_sample - curr_sample));
216  } else {
217  val = 0.0;
218  }
219  out(o, t) = val;
220  }
221  }
