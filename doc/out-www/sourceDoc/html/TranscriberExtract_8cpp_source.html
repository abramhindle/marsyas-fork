32 {
33  mrs_real highFreq = 5000.0;

35  MarSystem *net = mng.create("Series", "pitchNet");
36  net->addMarSystem(mng.create("ShiftInput", "sfi"));
37  net->addMarSystem(mng.create("PitchPraat", "pitch"));
38  if (rvSink != NULL)
39  net->addMarSystem(rvSink);
40 
41  // yes, this is the right way around (lowSamples<-highFreq)
42  net->updControl("PitchPraat/pitch/mrs_natural/lowSamples",
43  hertz2samples(highFreq, srate) );
44  net->updControl("PitchPraat/pitch/mrs_natural/highSamples",
45  hertz2samples(lowFreq, srate) );
46 
47  // The window should be just long enough to contain three periods
48  // (for pitch detection) of MinimumPitch.
49  mrs_real windowSize = 3.0/lowFreq*srate;
50  net->updControl("mrs_natural/inSamples", 512);
51  net->updControl("ShiftInput/sfi/mrs_natural/winSize",
52  powerOfTwo(windowSize));
53 
54  return net;

150 TranscriberExtract::getPitchesFromRealvecSink(MarSystem* rvSink,
151  const mrs_real srate)
152 {
153  realvec data = rvSink->getctrl("mrs_realvec/data")->to<mrs_realvec>();
154  rvSink->updControl("mrs_bool/done", true);
155 
156  realvec pitchList(data.getSize()/2);
157  mrs_real pitchOutput;
158  for (mrs_natural i=0; i<pitchList.getSize(); ++i)
159  {
160  // on linux (but not OSX), we have pitchOutput of 0.5 if the pitch
161  // detection can't decide on a pitch.
162  pitchOutput = data(2*i+1);
163  if (pitchOutput > 1)
164  pitchList(i) = samples2hertz( pitchOutput, srate);
165  else
166  pitchList(i) = 0;
167  }
168  return pitchList;
169 }

187 mrs_real
188 TranscriberExtract::getNormalizingGain(const std::string audioFilename)
189 {
190  mrs_real maxVal = 0.0;
191 
192  MarSystem* pnet = mng.create("Series", "pnet");
193  addFileSource(pnet, audioFilename);
194  // forces Marsyas to write to processedData
195  pnet->addMarSystem(mng.create("Gain", "null"));
196 
197  while ( pnet->getctrl("mrs_bool/hasData")->to<mrs_bool>() )
198  {
199  pnet->tick();
200  const realvec& processedData =
201  pnet->getctrl("SoundFileSource/src/mrs_realvec/processedData")->to<mrs_realvec>();
202  for (mrs_natural i=0; i< processedData.getSize(); ++i)
203  {
204  mrs_real val = fabs(processedData(i));
205  if (val > maxVal)
206  maxVal = val;
207  }
208  }
209 
210  delete pnet;
211  return 1.0/maxVal;
212 }
