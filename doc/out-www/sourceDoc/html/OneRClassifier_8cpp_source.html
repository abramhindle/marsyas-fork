139 //Sorts the data table on this attribute and executes the OneR algorithm.
140 OneRClassifier::OneRRule *OneRClassifier::newRule(mrs_natural attr, mrs_natural nClasses)
141 {
142  //create the counting variables
143  vector<mrs_natural> classifications(instances_.size());
144  vector<mrs_real> breakpoints(instances_.size());
145  vector<mrs_natural> counts(nClasses);
146 
147  //set correct count to 0
148  mrs_natural correct = 0;
149  mrs_natural lastInstance = instances_.size();
150 
151  //Sort the data table for this attribute
152  instances_.Sort(attr);
153 
154  mrs_natural ii = 0;
155  mrs_natural cl = 0; //index of next bucket to create
156  mrs_natural it = 0;
157 
158  //scan thru all rows in table
159  while(ii < lastInstance)
160  {
161  //zero the current counts
162  for(mrs_natural jj=0; jj<(mrs_natural)counts.size(); jj++) counts[jj]=0;
163  do
164  {//fill it until is has enough of the majority class
165  it = instances_.GetClass(++ii);
166  counts[it]++;
167  }while(counts[it] < minBucketSize_ && ii < lastInstance);
168 
169  //while class remains the same, keep on filling
170  while(ii < lastInstance && instances_.GetClass(ii) == it)
171  {
172  counts[it]++;
173  ++ii;
174  }//while
175 
176  //keep on while attr value is the same
177  while(ii < lastInstance && instances_.at(ii-1)->at(attr) == instances_.at(ii)->at(attr))
178  {
179  mrs_natural index = instances_.GetClass(ii++);
180  counts[index]++;
181  }//while
182 
183  for(mrs_natural jj=0; jj<nClasses; jj++)
184  {
185  if(counts[jj] > counts[it])
186  {
187  it = jj;
188  }//if
189  }//for jj
190 
191  if(cl > 0)
192  {//can we coalesce with previous class?
193  if(counts[classifications[cl-1]] == counts[it])
194  it = classifications[cl-1];
195 
196  if(it == classifications[cl-1])
197  cl--;
198  }//if
199 
200  correct += counts[it];
201  classifications[cl] = it;
202 
203  if(ii < lastInstance)
204  breakpoints[cl] = (((instances_.at(ii-1)->at(attr) + instances_.at(ii)->at(attr)) / 2.0));
205 
206  cl++;
207  }//while
208 
209  //create a new rule with cl branches
210  OneRRule *rule = new OneRRule(attr, cl, correct);
211  for(mrs_natural vv=0; vv<cl; vv++)
212  {
213  rule->getClassifications()[vv] = classifications[vv];
214  if(vv < (cl-1))
215  rule->getBreakpoints()[vv] = breakpoints[vv];
216 
217  }//for vv
218 
219  return rule;
220 }//newRule

222 //Build the classifier from the data table
223 void
224 OneRClassifier::Build(mrs_natural nClasses)
225 {
226  //make sure any previous rule is out
227  if(rule_!=NULL)
228  delete rule_;
229  rule_ = NULL;
230 
231  //scan through all the attributes(columns) of the table
232  for(mrs_natural enu = 0; enu < instances_.getCols()-1; enu++)
233  {
234  //construct a new rule for this attribute
235  OneRClassifier::OneRRule *r = newRule(enu, nClasses);
236 
237  //if a current rule does not exist or this new rule is better, replace old rule
238  if(!rule_ || r->getCorrect() > rule_->getCorrect())
239  {
240  if(rule_!=NULL)
241  delete rule_;
242 
243  rule_ = r;
244  }//if
245  }//for enu
246 }//Build

248 //Predict a class given a row of attribute data
249 mrs_natural
250 OneRClassifier::Predict(const realvec& in)
251 {
252  mrs_natural vv = 0;
253  mrs_real instValue = in(rule_->getAttr());
254 
255  //find the breakpoint whose value exceeds the attribute value.
256  while(vv < rule_->getnBreaks()-1 && instValue >= rule_->getBreakpoints()[vv])
257  {
258  vv++;
259  }//while
260 
261  //return the class for this prediction.
262  return rule_->getClassifications()[vv];
263 }//Predict
