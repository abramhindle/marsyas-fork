77 void
78 ParallelMatrixWeight::myProcess(realvec& in, realvec& out)
79 {
80  mrs_realvec weights = ctrl_weights_->to<mrs_realvec> ();
81  mrs_natural k,i,j,
82  numRows = weights.getRows (),
83  numCols = weights.getCols (),
84  intRows,
85  intCols;
86 
87  if (numRows == 0)
88  {
89  out.setval(0);
90  return;
91  }
92 
93  if (in.getRows () % numRows)
94  {
95  MRSWARN("ParallelMatrixWeight: dimension mismatch");
96  MRSASSERT(false);
97  out.setval(0);
98  return;
99  }
100 
101  intRows = in.getRows () / numRows,
102  intCols = in.getCols ();
103 
104  out = in;
105 
106  if (numCols == 1)
107  {
108  for (k = 0; k < numRows; k++)
109  {
110  mrs_real weight = weights(k);
111  for (i = 0; i < intRows; i++)
112  {
113  for (j = 0; j < intCols; j++)
114  {
115  out(k*intRows+i,j) *= weight;
116  }
117  }
118  }
119  }
120  else
121  {
122  if (in.getCols () % numCols || in.getRows () != numRows)
123  {
124  MRSWARN("ParallelMatrixWeight: dimension mismatch");
125  MRSASSERT(false);
126  out.setval(0);
127  return;
128  }
129 
130  out *= weights;
131  }
132 #ifdef MARSYAS_MATLAB
133 #ifdef MTLB_DBG_LOG
134  MATLAB_PUT(in, "in");
135  MATLAB_PUT(out, "out");
136  MATLAB_PUT(weights, "weights");
137  MATLAB_EVAL("figure(2);subplot(221),imagesc(in),colorbar;subplot(222),imagesc(out),colorbar;subplot(212),imagesc(weights),colorbar;");
138 #endif
139 #endif
140 }
