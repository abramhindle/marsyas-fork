59 void BlitOsc::myUpdate(MarControlPtr sender)
60 {
61  ap1.set_delay(1.9);
62  ap2.set_delay(1.3);
63 
64  frequency_ = (getctrl("mrs_real/frequency")->to<mrs_real>());
65  type_ = (getctrl("mrs_natural/type")->to<mrs_natural>());
66  noteon_ = (getctrl("mrs_bool/noteon")->to<mrs_bool>());
67  israte_ = (getctrl("mrs_real/israte")->to<mrs_real>());
68 
69  phase_ = 0;
70  inv_ = 1;
71 
72  switch (type_)
73  {
74  case 0: // Saw
75  std::cout << frequency_ << std::endl;
76  dc_ = frequency_/israte_;
77  break;
78  case 1: // Square
79  // The frequency has to be doubled to compensate for
80  // the frequency be being halved by the square wave
81  // being bipolar
82  frequency_ *= 2;
83  dc_ = 0;
84  break;
85  }
86 
87  // d is how many samples to delay
88  // because it is possible a fractional amount we split it
89  // into the integer part, and the fractional part.
90  mrs_real d = israte_/frequency_;
91  // N_ is the integer part.
92  N_ = (mrs_natural)floor(d);
93  // frac_ is the fractional part
94  frac_ = (d - N_);
95  // delay is used as an overflow counter for the
96  // fractional part.
97  delay_ = frac_;
98 
99  // no change to network flow
100  MarSystem::myUpdate(sender);
101 }

103 void BlitOsc::myProcess(realvec& in, realvec& out)
104 {
105  (void) in;
106  for (mrs_natural t = 0; t < inSamples_; t++)
107  {
108  if (phase_ >= N_ - 1)
109  {
110  phase_ = 0;
111 
112  // The amount of delay is incresed by 1 to optimize the allpass filter.
113  // N_ is compensated accordingly for this delay at the top of this if
114  // statement.
115  ap1.set_delay(delay_ + 1);
116 
117  switch (type_)
118  {
119  case 0: // Saw
120  out(0,t) = le(ap2(ap1(1)) - dc_);
121  break;
122  case 1: // Square
123  out(0,t) = le(ap2(ap1(1 * inv_)));
124  inv_ = (-inv_);
125  break;
126  }
127 
128  // This is the logic to tune the fractional part of the period.
129  delay_ += frac_;
130  if (delay_ >= 1)
131  {
132  delay_ -= 1;
133  phase_ = -1;
134  }
135  }
136  else
137  {
138  phase_++;
139  out(0,t) = le(ap2(ap1(0)) - dc_);
140  }
141  }
142 }
