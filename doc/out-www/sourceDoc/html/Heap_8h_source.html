37 /* This is heap is designed to sort pointers to objects.
38  The heap template requires a Type, a Comparator object for determining Type
39  ordering, and a Destructor object for destroying Type objects still in the
40  heap once the program finishes.
41 
42  Examples of Comparator and Destructor objects for Heap declaration:
43 
44  Heap<Object, Comp, Destr> h;
45 
46  class Comp {
47  public:
48  bool operator()(Object* a, Object* b) { return (a->m_id()) < (b->m_id()); }
49  };
50  class Destr {
51  public:
52  void operator()(Object* a) { delete(a); }
53  };
54 */

60 class Heap {
61 private:
62  class Node {
63  public:
64  Node* parent; Node* lchild; Node* rchild; // tree pointers

66  unsigned int m_id; // node id, for determining child type

75  o << "<" << s->m_id << "," << s->data << ",(";
76  if (s->parent==NULL) { o<<"0"; } else { o<<s->parent->m_id; } o << ",";

78  if (s->rchild==NULL) { o<<"x"; } else { o<<s->rchild->m_id; } o << ")>";
79  return o;

85  Node* first; Node* last; // first and last pointers in heap
86  // assigned to a node to identify root, l/r child.
87  // An empty tree has a count of 0, while the root node is always 1.
88  unsigned int node_count;
89  // name declaration for required Comparator object
90  Comparator cmp;
91 
92  Heap() { first=NULL; last=NULL; node_count=0; }
93  virtual ~Heap() {
94  while(first!=NULL) {
95  last=first->next;
96  // use the supplied Destructor object to delete the data
97  delete(first->data); delete(first);
98  first=last;
99  }
100  }
101 
102  bool empty() { return (node_count==0); };
103 
104  Type* top() {
105  // on empty heap throw a const char* exception, specified in the contract
106  if (first==NULL) { throw "Heap::top() empty heap exception."; }
107  else { return first->data; }
108  };
109  Type* pop() {
110  // on empty heap throw a const char* exception, specified in the contract
111  if (first==NULL) { throw "Heap::pop() empty heap exception."; }
112  // save top data
113  Type* data = first->data;
114  // if it's the root then clear the heap
115  if (is_root(last)) {
116  delete(last); first=NULL; last=NULL; node_count=0;
117  }
118  else {
119  // swap last element data into top position
120  first->data = last->data;
121  // extricate the node from its parent
122  if (is_lchild(last)) { last->parent->lchild=NULL; }
123  else { last->parent->rchild=NULL; }
124  // remove the last node and update the pointer to the new last
125  last=last->prev; delete(last->next); last->next=NULL;
126  // bubble down
127  Node* f = first;
128  while (1) {
129  Node* c = f->lchild;
130  // if lchild of c==NULL then c cannot be bubbled down further
131  if (c==NULL) { break; }
132  // make f point to the smallest of c's children
133  if (f->rchild!=NULL && (cmp(f->rchild->data,c->data))) { c=f->rchild; }
134  // use the template required Comparator to compare if the
135  // smallest child of c is less, if not leave
136  if (cmp(f->data,c->data)) { break; }
137  // swap data
138  Type* sw=c->data; c->data=f->data; f->data=sw;
139  // track the bubbling node
140  f=c;
141  }
142  // update node_count ensuring it never goes below 0
143  node_count = (node_count>0) ? node_count-1 : 0;
144  }
145  return data;
146  };
147  // push
148  void push(Type* data) {
149  // could throw an exception here...hmm
150  if (data==NULL) { return; }
151  node_count++;
152 
153  Node* n = new Node(node_count,data);
154  if (first==NULL) {
155  first=n; last=n;
156  }
157  else {
158  // add node to end of list
159  last->next=n; n->prev=last;
160  // insert node into tree and update parent and parent_children
161  if (is_root(last)) {
162  n->parent=last;
163  last->lchild = n;
164  }
165  else if (is_lchild(last)) {
166  n->parent=last->parent;
167  last->parent->rchild=n;
168  }
169  else {
170  n->parent=last->parent->next;
171  last->parent->next->lchild=n;
172  }
173  last=n;
174  // bubble up if needed
175  while (!is_root(n) && cmp(n->data,n->parent->data)) {
176  Type* sw=n->data; n->data=n->parent->data; n->parent->data=sw;
177  n = n->parent;
178  }
179  }
180  };
181  friend std::ostream& operator<<(std::ostream& o, Heap& s) {
182  Node* c = s.first;
183  while (c!=NULL) { o << c; c = c->next; } o << "\n";
184  return o;
185  };
