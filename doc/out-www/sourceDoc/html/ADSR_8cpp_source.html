113 
114 
115  if(noteon_ || nton_)
116  {
117  this->setctrl("mrs_real/nton",0.0);

133 void
134 ADSR::myProcess(realvec& in, realvec& out)
135 {
136  mrs_natural o,t;
137  for (o = 0; o < inObservations_; o++)
138  {
139  for (t = 0; t < inSamples_; t++)
140  {
141  switch (state_)
142  {
143  case 1://attack
144  value_ += aRate_;
145  if (value_ >= target_)
146  {
147  value_ = target_;
148  rate_ = dRate_;
149  target_ = susLevel_;
150  state_ = 2;
151  }
152  break;
153  case 2://decay
154  value_ -= dRate_;
155  if (value_ <= susLevel_)
156  {
157  value_ = susLevel_;
158  rate_ = 0.0;
159  state_ = 3;
160  }
161  break;
162  case 4://release
163  value_ -= rRate_;
164  if (value_ <= 0.0)
165  {
166  value_ = 0.0;
167  state_ = 5;//done
168  }
169  }//switch
170 
171  if ( !bypass_ )
172  {
173  out(o,t) = value_* in(o,t);
174  }
175  else
176  {
177  out(o,t) = value_;
178  }
179  }//for
180  }//for
