25 /*
26 * bitreverse places real array x containing N/2 complex values
27 * into bit-reversed order
28 */
29 
30 void
31 fft::bitreverse(mrs_real x[], int N )
32 {
33  mrs_real rtemp, itemp ;
34  int i, j, m ;
35  for ( i = j = 0 ; i < N ; i += 2, j += m ) {
36  if ( j > i ) {
37  rtemp = x[j] ; itemp = x[j+1] ; /* complex exchange */
38  x[j] = x[i] ; x[j+1] = x[i+1] ;
39  x[i] = rtemp ; x[i+1] = itemp ;
40  }
41  for ( m = N>>1 ; m >= 2 && j >= m ; m >>= 1 )
42  j -= m ;
43  }
44 }

48 /*
49 * COMPUTES A 2*N SIZED FOURIER TRANSFORM
50 *
51 * If forward is true, rfft replaces 2*N real data points in x with
52 * N complex values representing the positive frequency half of their
53 * Fourier spectrum, with x[1] replaced with the real part of the Nyquist
54 * frequency value. If forward is false, rfft expects x to contain a
55 * positive frequency spectrum arranged as before, and replaces it with
56 * 2*N real values. N MUST be a power of 2.
57 */
58 void
59 fft::rfft( mrs_real x[], int N, int forward )
60 {
61  mrs_real c1, c2, h1r, h1i, h2r, h2i, wr, wi, wpr, wpi, temp, theta ;
62  mrs_real xr, xi ;
63  int i, i1, i2, i3, i4, N2p1 ;
64  theta = PI/N ;
65  wr = 1. ;
66  wi = 0. ;
67  c1 = 0.5 ;
68  if ( forward )
69  {
70  c2 = -0.5 ;
71  cfft( x, N, forward ) ;
72  xr = x[0] ;
73  xi = x[1] ;
74  }
75  else
76  {
77  c2 = 0.5 ;
78  theta = -theta ;
79  xr = x[1] ;
80  xi = 0. ;
81  x[1] = 0. ;
82  }
83  wpr = (mrs_real)(-2.*pow( sin( 0.5*theta ), 2. ));
84  wpi = sin( theta ) ;
85  N2p1 = (N<<1) + 1 ;
86  for ( i = 0 ; i <= N>>1 ; ++i )
87  {
88  i1 = i<<1 ;
89  i2 = i1 + 1 ;
90  i3 = N2p1 - i2 ;
91  i4 = i3 + 1 ;
92  if ( i == 0 ) {
93  h1r = c1*(x[i1] + xr ) ;
94  h1i = c1*(x[i2] - xi ) ;
95  h2r = -c2*(x[i2] + xi ) ;
96  h2i = c2*(x[i1] - xr ) ;
97  x[i1] = h1r + wr*h2r - wi*h2i ;
98  x[i2] = h1i + wr*h2i + wi*h2r ;
99  xr = h1r - wr*h2r + wi*h2i ;
100  xi = -h1i + wr*h2i + wi*h2r ;
101  }
102  else {
103  h1r = c1*(x[i1] + x[i3] ) ;
104  h1i = c1*(x[i2] - x[i4] ) ;
105  h2r = -c2*(x[i2] + x[i4] ) ;
106  h2i = c2*(x[i1] - x[i3] ) ;
107  x[i1] = h1r + wr*h2r - wi*h2i ;
108  x[i2] = h1i + wr*h2i + wi*h2r ;
109  x[i3] = h1r - wr*h2r + wi*h2i ;
110  x[i4] = -h1i + wr*h2i + wi*h2r ;
111  }
112  wr = (temp = wr)*wpr - wi*wpi + wr ;
113  wi = wi*wpr + temp*wpi + wi ;
114  }
115  if ( forward )
116  x[1] = xr ;
117  else
118  cfft( x, N, forward ) ;
119 }

122 /*
123 * cfft replaces real array x containing NC complex values
124 * (2*NC real values alternating real, imagininary, etc.)
125 * by its Fourier transform if forward is true, or by its
126 * inverse Fourier transform if forward is false, using a
127 * recursive Fast Fourier transform method due to Danielson
128 * and Lanczos. NC MUST be a power of 2.
129 */
130 
131 void
132 fft::cfft(mrs_real x[], int NC, int forward )
133 {
134  mrs_real wr, wi, wpr, wpi, theta, scale ;
135  int mmax, ND, m, i, j, delta ;
136  ND = NC<<1 ;
137  bitreverse( x, ND ) ;
138  for ( mmax = 2 ; mmax < ND ; mmax = delta ) {
139  delta = mmax<<1 ;
140  theta = TWOPI/( forward? mmax : -mmax ) ;
141  wpr = (mrs_real)(-2.*pow( sin( 0.5*theta ), 2. ));
142  wpi = sin( theta ) ;
143  wr = 1. ;
144  wi = 0. ;
145  for ( m = 0 ; m < mmax ; m += 2 ) {
146  register mrs_real rtemp, itemp ;
147  for ( i = m ; i < ND ; i += delta )
148  {
149  j = i + mmax ;
150  rtemp = wr*x[j] - wi*x[j+1] ;
151  itemp = wr*x[j+1] + wi*x[j] ;
152  x[j] = x[i] - rtemp ;
153  x[j+1] = x[i+1] - itemp ;
154  x[i] += rtemp ;
155  x[i+1] += itemp ;
156  }
157  wr = (rtemp = wr)*wpr - wi*wpi + wr ;
158  wi = wi*wpr + rtemp*wpi + wi ;
159  }
160  }
161  /*
162  * scale output
163  */
164 
165  scale = forward ? (mrs_real)1.0/ND : (mrs_real)2.0 ;
166  { register mrs_real *xi=x, *xe=x+ND ;
167  while ( xi < xe )
168  *xi++ *= scale ;
169  }
170 
171 }
