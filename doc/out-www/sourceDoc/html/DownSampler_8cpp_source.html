56 void
57 DownSampler::myUpdate(MarControlPtr sender)
58 {
59  MRSDIAG("DownSampler.cpp - DownSampler:myUpdate");
60 
61  // Use the default MarSystem setup with equal input/output stream format ...
62  MarSystem::myUpdate(sender);
63 
64  // ... but override the onSamples and osrate settings based on the
65  // downsample factor.
66  mrs_natural factor = ctrl_factor_->to<mrs_natural>();
67  ctrl_osrate_->setValue(ctrl_israte_->to<mrs_real>() / factor, NOUPDATE);
68  // When inSamples is divisible by factor, the number of output samples
69  // onSamples is just the integer division inSamples / factor.
70  // Otherwise, onSamples is the biggest integer obeying
71  // factor * (onSamples - 1) < inSamples
72  // which is ceil(inSamples / factor).
73  mrs_natural onSamples = (mrs_natural) ceil((mrs_real)(ctrl_inSamples_->to<mrs_natural>()) / factor);
74  ctrl_onSamples_->setValue(onSamples, NOUPDATE);
75 }

77 void
78 DownSampler::myProcess(realvec& in, realvec& out)
79 {
80  mrs_natural t,o;
81  mrs_natural factor = ctrl_factor_->to<mrs_natural>();
82 
83  // Only copy samples at multiples of factor.
84  for (o=0; o < inObservations_; o++)
85  {
86  for (t = 0; t < onSamples_; t++)
87  {
88  out(o, t) = in(o, t * factor);
89  }
90  }
91 }
