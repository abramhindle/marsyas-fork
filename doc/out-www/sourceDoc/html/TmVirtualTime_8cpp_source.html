50 TmVirtualTime::setSource(MarSystem* ms)
51 {
52  // start at 0
53  previous_tick_interval_=0;
54  read_src_=ms;
55  error_term_=0.0;
56  if(read_src_!=NULL) {
57  nsamples_=read_src_->getctrl("mrs_natural/onSamples");
58  srate_=read_src_->getctrl("mrs_real/osrate");
59  }

62 mrs_natural
63 TmVirtualTime::readTimeSrc()
64 {
65  if (read_src_==NULL) {
66  MRSWARN("TmVirtualTime::readTimeSrc() time source is not defined.");
67  return 0;
68  }
69  // this is the width of the last tick - elapsed time since last tick
70  mrs_natural ret = previous_tick_interval_;
71  mrs_real srate = srate_->to<mrs_real>();
72  if (srate<1.0)
73  return 0;
74  mrs_real interval_width = nsamples_->to<mrs_natural>() / srate;
75  mrs_real microseconds = (interval_width * 1000000.0) + error_term_;
76  error_term_ = microseconds - ((long)microseconds);
77 // mrs_natural count = (mrs_natural)microseconds;
78 // cout << "us=" << ((mrs_natural)microseconds) << endl;
79  previous_tick_interval_ = (mrs_natural)microseconds;
80  return ret;
81 }
