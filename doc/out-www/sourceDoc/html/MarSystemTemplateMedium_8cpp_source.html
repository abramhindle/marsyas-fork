58 void
59 MarSystemTemplateMedium::addControls()
60 {
61  // Add any specific controls needed by this MarSystem.
62 
63  // Let's start by adding a dummy control (to which we will not use
64  // a "pointer", just to illustrate the "traditional", yet
65  // not so efficient way of using controls)
66  addctrl("mrs_bool/dummy", false);
67  // in this case this control should have state, since
68  // other controls will depend on it.
69  // (i.e. any change to it will call MarSystem::update()
70  // which then calls myUpdate(MarControlPtr sender))
71  setctrlState("mrs_bool/dummy", true);
72 
73  // if a "pointer" to a control is to be used (for efficiency purposes
74  // - see myProcess() bellow), it should be passed as the last
75  // argument to addctrl()
76  addctrl("mrs_real/gain", 1.0, ctrl_gain_EXAMPLE_);
77  // IMPORTANT NOTE:
78  // in the above case, since the control value is supposed to
79  // be a mrs_real, the default value also has to be a mrs_real!
80  // if not (e.g. initialiting with "1" instead of "1.0"), the
81  // control will in fact have a mrs_natural value despite of
82  // the "mrs_real/..." name.
83 
84  addctrl("mrs_natural/repeats", 1, ctrl_repeats_EXAMPLE_);
85  // if we have a "pointer" to a control, we can set its state
86  // in a different and more efficient way
87  ctrl_repeats_EXAMPLE_->setState(true);
88 }

90 void
91 MarSystemTemplateMedium::myUpdate(MarControlPtr sender)
92 {
93  MRSDIAG("MarSystemTemplateMedium.cpp - MarSystemTemplateMedium:myUpdate");
94 
95 
96  MarSystem::myUpdate(sender);
97  // As an example, this MarSystem will output a nr of repeated input frames
98  //
99  // The traditional (and still valid), but inefficient way to set controls:
100  //
101  //setctrl("mrs_natural/onObservations", getctrl("mrs_natural/inObservations"));
102  //setctrl("mrs_real/osrate", getctrl("mrs_real/israte"));
103  //setctrl("mrs_string/onObsNames", getctrl("mrs_string/inObsNames"));
104  //...
105  //
106  // more efficient way to set the value of a control
107  // (if we have a "pointer" to it - see addControls(), above).
108  // setValue() always calls update() if the control in cause has state.
109  // To avoid calling update() (and avoid recursion since we are already
110  // inside update()/myUpdate(MarControlPtr sender)), setValue() should be called with the
111  // NOUPDATE flag, as bellow:
112 /*
113  ctrl_onObservations_->setValue(ctrl_inObservations_, NOUPDATE);
114  ctrl_osrate_->setValue(ctrl_israte_, NOUPDATE);
115  ctrl_onObsNames_->setValue(ctrl_inObsNames_, NOUPDATE);
116 */
117  // here it's possible to see how the "repeats" control
118  // affects the onSamples control (i.e. output number of samples)
119  //
120  // IMPORTANT: Since this changes the onSamples control, if this
121  // MarSystem is inside a Composite, the Composite::update() must be
122  // called afterwards in the main code, otherwise the size of the
123  // corresponding slice will not be updated accordingly!
124  // (see marsyasTests.cpp >> test_MarControls() )
125  //
126  ctrl_onSamples_->setValue(ctrl_inSamples_ * ctrl_repeats_EXAMPLE_, NOUPDATE);
127 
128  // NOTE:
129  // see Gain.cpp for some more info about output parameter configuration.
130 }

132 void
133 MarSystemTemplateMedium::myProcess(realvec& in, realvec& out)
134 {
135  mrs_natural t,o;
136  // FIXME Variable (rep) appears to be unused.
137  // static mrs_natural rep = 1;
138  static realvec vec;
139 
140  const mrs_real& gainValue = ctrl_gain_EXAMPLE_->to<mrs_real>();
141  // this is equivalent (although slightly more efficient) to:
142  //
143  // mrs_real& gainValue = ctrl_gain_EXAMPLE_->to<mrs_real>(); // ::toReal() calls ::to<mrs_real>()
144  //
145  // This reference will not allow writing directly to the control, but
146  // avoids a copy (which can hurt if the control is a big realvec)
147  // and even if by some means the control value is modified elsewhere (e.g. by a
148  // different thread), it's always in sync with the actual control value.
149 
150  // There may be cases where is more adequate to get a copy of the control value,
151  // so it does not change event if the actual control value is changed elsewhere:
152  mrs_natural repeats = ctrl_repeats_EXAMPLE_->to<mrs_natural>();
153 
154  for (o=0; o < inObservations_; o++)
155  for (t = 0; t < inSamples_; t++)
156  for(mrs_natural r= 0; r < repeats; ++r)
157  out(o,t+r*inSamples_) = gainValue * in(o,t);
158 
159  // Just as an example, let's assume we want to write to a control
160  // (e.g. some weird amplitude modulation effect), this is the way
161  // to do it:
162  mrs_real g = (((mrs_natural)(gainValue*10)+1)%10)/10.0;
163  ctrl_gain_EXAMPLE_->setValue(g);
164 }
