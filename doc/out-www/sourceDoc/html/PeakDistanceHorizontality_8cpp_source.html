92 void
93 PeakDistanceHorizontality::myProcess(realvec& in, realvec& out)
94 {
95  mrs_natural i;
96  const mrs_natural numInputs = getctrl ("mrs_natural/numInputs")->to<mrs_natural>();
97  const mrs_realvec isHoriz = ctrl_horizvert_->to<mrs_realvec>();
98  const mrs_real range[2] = {ctrl_rangeX_->to<mrs_real>(), ctrl_rangeY_->to<mrs_real>()};
99 
100  out = in;
101 
102  MRSASSERT(range[0] > 0 && range[1] > 0);
103  if (isHoriz.getSize () != numInputs)
104  {
105  MRSWARN("PeakDistanceHorizontality: dimension mismatch");
106  MRSASSERT(false);
107  out.setval(0);
108  return;
109  }
110 
111  if (getctrl("mrs_bool/bypass")->to<mrs_bool>())
112  {
113  weights_.setval(1.);
114  setctrl ("mrs_realvec/weights", weights_);
115  return;
116  }
117 
118  for (i = 0; i < inSamples_; i++)
119  {
120  for (mrs_natural j = i; j < inSamples_; j++)
121  {
122  mrs_natural k;
123  mrs_real horizontality = ComputeHorizontality ( std::abs(in(1,i)-in(1,j))/range[0],
124  std::abs(in(0,i)-in(0,j))/range[1]),
125  norm = 0;
126 
127  for (k = 0; k < numInputs; k++)
128  {
129  mrs_real weight = horizontality;
130 
131  if (abs(isHoriz(k) - 2) < kIdentityThresh)
132  weight = .5; // input is both horizontal and vertical
133  else if (abs(isHoriz(k)) < kIdentityThresh)
134  weight = 1.-weight; // input is vertical
135 
136  norm += weight;
137  weights_(k*inSamples_ + i, j) = weight;
138  weights_(k*inSamples_ + j, i) = weight; // symmetry
139  }
140  if (norm != 0)
141  norm = 1./norm;
142  for (k = 0; k < numInputs; k++)
143  {
144  weights_(k*inSamples_ + i, j) *= norm;
145  if (i != j)
146  weights_(k*inSamples_ + j, i) *= norm; // symmetry
147  }
148  }
149  }
150  setctrl ("mrs_realvec/weights", weights_);
151 #ifdef MARSYAS_MATLAB
152 #ifdef MTLB_DBG_LOG
153  MATLAB_PUT(weights_, "weights");
154  MATLAB_EVAL("figure(1);imagesc(weights),colorbar;");
155 #endif
156 #endif
157 }

159 mrs_real
160 PeakDistanceHorizontality::ComputeHorizontality(mrs_real scaledDiffX, mrs_real scaledDiffY)
161 {
162 
163  if (scaledDiffX == 0)
164  {
165  if (scaledDiffY == 0)
166  return .5;
167  else
168  return 0.;
169  }
170  else if (scaledDiffY == 0)
171  return 1.;
172 
173  mrs_real res = scaledDiffX / std::sqrt(scaledDiffX*scaledDiffX + scaledDiffY*scaledDiffY);
174 
175  return res*res;
176 }
