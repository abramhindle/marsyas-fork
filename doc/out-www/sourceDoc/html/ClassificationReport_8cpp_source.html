255 summaryStatistics ClassificationReport::computeSummaryStatistics(const realvec& mat)
256 {
257  MRSASSERT(mat.getCols()==mat.getRows());
258 
259  summaryStatistics stats;
260 
261  mrs_natural size = mat.getCols();
262 
263  vector<mrs_natural>rowSums(size);
264  for(int ii=0; ii<size; ++ii) rowSums[ii] = 0;
265  vector<mrs_natural>colSums(size);
266  for(int ii=0; ii<size; ++ii) colSums[ii] = 0;
267  mrs_natural diagonalSum = 0;
268 
269  mrs_natural instanceCount = 0;
270  for(mrs_natural row=0; row<size; row++)
271  {
272  for(mrs_natural col=0; col<size; col++)
273  {
274  mrs_natural num = (mrs_natural)mat(row,col);
275  instanceCount += num;
276 
277  rowSums[row] += num;
278  colSums[col] += num;
279 
280  if(row==col)
281  diagonalSum += num;
282  }
283  }
284  //printf("row1 sum:%d\n",rowSums[0]);
285  //printf("row2 sum:%d\n",rowSums[1]);
286  //printf("col1 sum:%d\n",colSums[0]);
287  //printf("col2 sum:%d\n",colSums[1]);
288  //printf("diagonal sum:%d\n",diagonalSum);
289  //printf("instanceCount:%d\n",instanceCount);
290 
291  mrs_natural N = instanceCount;
292  mrs_natural N2 = (N*N);
293  stats.instances = instanceCount;
294  stats.correctInstances = diagonalSum;
295 
296  mrs_natural sum = 0;
297  for(mrs_natural ii=0; ii<size; ++ii)
298  {
299  sum += (rowSums[ii] * colSums[ii]);
300  }
301  mrs_real PE = (mrs_real)sum / (mrs_real)N2;
302  mrs_real PA = (mrs_real)diagonalSum / (mrs_real)N;
303  stats.kappa = (PA - PE) / (1.0 - PE);
304 
305  mrs_natural not_diagonal_sum = instanceCount - diagonalSum;
306  mrs_real MeanAbsoluteError = (mrs_real)not_diagonal_sum / (mrs_real)instanceCount;
307  //printf("MeanAbsoluteError:%f\n",MeanAbsoluteError);
308  stats.meanAbsoluteError = MeanAbsoluteError;
309 
310  mrs_real RootMeanSquaredError = sqrt(MeanAbsoluteError);
311  //printf("RootMeanSquaredError:%f\n",RootMeanSquaredError);
312  stats.rootMeanSquaredError = RootMeanSquaredError;
313 
314  mrs_real RelativeAbsoluteError = (MeanAbsoluteError / 0.5) * 100.0;
315  //printf("RelativeAbsoluteError:%f%%\n",RelativeAbsoluteError);
316  stats.relativeAbsoluteError = RelativeAbsoluteError;
317 
318  mrs_real RootRelativeSquaredError = (RootMeanSquaredError / (0.5)) * 100.0;
319  //printf("RootRelativeSquaredError:%f%%\n",RootRelativeSquaredError);
320  stats.rootRelativeSquaredError = RootRelativeSquaredError;
321 
322  return stats;
323 }//computeSummaryStatistics
