76 void Pitch2Chroma::UpdatePitchToNoteTransform()
77 {
78  // Define pitch to note transformation
79  PitchToNoteTransform_.create(NrOfNotes_,inObservations_);
80  StartAndEndIndex_.create(NrOfNotes_,2);
81 
82  mrs_real theLOGFc = LowestPitch_;
83  mrs_real theLOGFStep = pow(2.,1./(mrs_real)NotesPerOctave_);
84  mrs_real theLinFStep = SampleRate_/(2.f*(mrs_real)inObservations_);
85  for (int i=0; i<NrOfNotes_; ++i)
86  {
87  // Define begin and end frequency
88  mrs_real theLOGFb = theLOGFc/sqrt(theLOGFStep);
89  mrs_real theLOGFe = theLOGFc*sqrt(theLOGFStep);
90 
91  // Convert begin and end frequency to FFT index
92  StartAndEndIndex_(i,0) = max((int)ceil(theLOGFb/theLinFStep),0);
93  StartAndEndIndex_(i,1) = min((int)floor(theLOGFe/theLinFStep),(int)inObservations_-1);
94 
95  for (int j=(mrs_natural)StartAndEndIndex_(i,0); j<=(mrs_natural)StartAndEndIndex_(i,1); j++)
96  {
97  double theBinFreq = (mrs_real)j*theLinFStep;
98  if (theBinFreq <= theLOGFc)
99  // Ramp up
100  PitchToNoteTransform_(i,j) = (theBinFreq-theLOGFb)/(theLOGFc-theLOGFb);
101  else
102  // Ramp down
103  PitchToNoteTransform_(i,j) = 1.-(theBinFreq-theLOGFc)/(theLOGFe-theLOGFc);
104  }
105 
106  theLOGFc *= theLOGFStep;
107  }
