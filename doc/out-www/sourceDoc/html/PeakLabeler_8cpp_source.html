61 void
62 PeakLabeler::myProcess(realvec& in, realvec& out)
63 {
64  out = in;
65 
66  peakView outPeakView(out);
67 
68  mrs_natural maxNumPeaks = outPeakView.getFrameMaxNumPeaks();
69  const realvec& peakLabels = ctrl_peakLabels_->to<mrs_realvec>(); //reading unlocked and linked control!!! --> should use MarControlAccessor [TODO][!]
70 
71  if((mrs_natural)peakLabels.getSize() != outPeakView.getTotalNumPeaks())
72  {
73  MRSERR("PeakLabeler::myProcess - peakLabels control and input peaks number mismatch! Labeling not performed!");
74  }
75  else
76  {
77  //fill in the peak labeling using the info
78  //from the peakLabels control (if any)
79  labelIdx_ = 0;
80  for(mrs_natural f=0; f < outPeakView.getNumFrames(); ++f)
81  {
82  for(mrs_natural p=0; p < outPeakView.getFrameNumPeaks(f); ++p)
83  {
84 // outPeakView(p, peakView::pkGroup, f) = peakLabels(labelIdx_); // assignment operator doesn't seem to work
85  out(p + maxNumPeaks*peakView::pkGroup, f) = peakLabels(labelIdx_);
86  labelIdx_++;
87  }
88  }
89  }
