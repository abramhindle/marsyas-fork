81  // find minimums_ and maximums_
82  for(vector<vector<mrs_real>*>::const_iterator citer = this->begin(); citer!=this->end(); citer++)
83  {
84  const vector<mrs_real> *row = (*citer);
85  int ii;

87  {
88  if (row->at(ii) > maximums_(ii))

151 mrs_natural WekaData::partition(mrs_natural attIndex, mrs_natural l, mrs_natural r)
152 {
153  mrs_real pivot = this->at((l+r)/2)->at(attIndex);
154  while (l < r)
155  {
156  while ((this->at(l)->at(attIndex) < pivot) && (l < r))
157  {
158  l++;
159  }//while
160 
161  while ((this->at(r)->at(attIndex) > pivot) && (l < r))
162  {
163  r--;
164  }//while
165 
166  if (l < r)
167  {
168  swapRows(l, r);
169  l++;
170  r--;
171  }//if
172  }
173  if ((l == r) && (this->at(r)->at(attIndex) > pivot))
174  {
175  r--;
176  } //if
177 
178  return r;
179 }//partition
