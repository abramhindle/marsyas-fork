71 void RunningStatistics::myUpdate(MarControlPtr sender)
72 {
73  (void) sender; //suppress warning of unused parameter(s)
74  MRSDIAG("RunningStatistics.cpp - RunningStatistics:myUpdate");
75 
76  // Update caches of control values.
77  enable_mean_ = ctrl_enable_mean_->to<mrs_bool> ();
78  enable_stddev_ = ctrl_enable_stddev_->to<mrs_bool> ();
79  enable_skewsness_ = ctrl_enable_skewness_->to<mrs_bool> ();
80 
81  mrs_natural fanout = (mrs_natural) enable_mean_
82  + (mrs_natural) enable_stddev_ + (mrs_natural) enable_skewsness_;
83 
84  // Set the output slice format and rate based on the
85  // input slice format and rate.
86  ctrl_onSamples_->setValue((mrs_natural) 1, NOUPDATE);
87  onObservations_ = fanout * ctrl_inObservations_->to<mrs_natural> ();
88  ctrl_onObservations_->setValue(onObservations_, NOUPDATE);
89  ctrl_osrate_->setValue(ctrl_israte_, NOUPDATE);
90 
91  // Update the observation names.
92  mrs_string inObsNames = ctrl_inObsNames_->to<mrs_string> ();
93  mrs_string onObsNames("");
94  if (enable_mean_)
95  {
96  onObsNames += obsNamesAddPrefix(inObsNames, "RunningMean_");
97  }
98  if (enable_stddev_)
99  {
100  onObsNames += obsNamesAddPrefix(inObsNames, "RunningStddev_");
101  }
102  if (enable_skewsness_)
103  {
104  onObsNames += obsNamesAddPrefix(inObsNames, "RunningSkewness_");
105  }
106 
107  ctrl_onObsNames_->setValue(onObsNames, NOUPDATE);
108 
109  // Allocate and initialize the buffers and counters.
110  sumxBuffer_.stretch(onObservations_, 1);
111  sumx2Buffer_.stretch(onObservations_, 1);
112  sumx3Buffer_.stretch(onObservations_, 1);
113  // Initialize the buffers.
114  clear();

126 void RunningStatistics::myProcess(realvec& in, realvec& out)
127 {
128  mrs_natural o,t;
129  if (ctrl_clear_->to<mrs_bool>() || ctrl_clearPerTick_->to<mrs_bool>()) {
130  clear();
131  }
132 
133  // Update the sample counter.
134  samplecounter_ += inSamples_;
135  // Temporary variables for mean, stddev, ...
136  mrs_real mean, var, stddev, skewness;
137  // Temporary helper variable.
138  mrs_real tmp;
139 
140  // Precalculate offsets where to put stddev and skewness in output.
141  mrs_natural o_offset_stdev = (mrs_natural) enable_mean_ * inObservations_;
142  mrs_natural o_offset_skewness = o_offset_stdev
143  + (mrs_natural) enable_stddev_ * inObservations_;
144 
145  for (o = 0; o < inObservations_; o++)
146  {
147  // Gather the sums.
148  for (t = 0; t < inSamples_; t++)
149  {
150  sumxBuffer_(o, 0) += in(o, t);
151  sumx2Buffer_(o, 0) += (tmp = in(o, t) * in(o, t));
152  sumx3Buffer_(o, 0) += (tmp * in(o, t));
153  }
154  // Calculate the statistics.
155  mean = sumxBuffer_(o, 0) / samplecounter_;
156  var = sumx2Buffer_(o, 0) / samplecounter_ - mean * mean;
157  stddev = sqrt(var);
158  skewness = (sumx3Buffer_(o, 0) / samplecounter_ - 3 * mean * var - mean
159  * mean * mean);
160  skewness = var > 0.0 ? skewness / (var * stddev) : 0.0;
161 
162  // Store in output slice.
163  if (enable_mean_)
164  {
165  out(o, 0) = mean;
166  }
167  if (enable_stddev_)
168  {
169  out(o + o_offset_stdev, 0) = stddev;
170  }
171  if (enable_skewsness_)
172  {
173  out(o + o_offset_skewness, 0) = skewness;
174  }
175 
176  }
177 }
