141 void
142 Filter::myProcess(realvec& in, realvec& out)
143 {
144  //checkFlow(in,out);
145 
146  mrs_natural i,j,c;
147  mrs_natural size = in.getCols();
148  mrs_natural stateSize = state_.getCols();
149  mrs_natural channels = in.getRows();
150 
151  mrs_real gain = getctrl("mrs_real/fgain")->to<mrs_real>();
152 
153  // State array holds the various delays for the difference equation
154  // of the filter. Similar implementation as described in the manual
155  // for MATLAB Signal Processing Toolbox. State corresponds to
156  // the z, num_coefs to the b and denom_coefs to the a vector respectively
157  // in_window is the input x(n) and out_window is the output y(n)
158 
159  //dcoeffs_/=10;
160 
161 
162  // state_.setval(0);
163  if (norder_ == dorder_){
164  for (c = 0; c < channels; ++c) {
165  for (i = 0; i < size; ++i){
166  out(c,i) = ncoeffs_(0) * in(c,i) + state_(c,0);
167  for (j = 0; j < stateSize - 1; j++)
168  {
169  state_(c,j) = ncoeffs_(j+1) * in(c,i) + state_(c,j+1) - dcoeffs_(j+1) * out(c,i);
170  }
171  state_(c,stateSize - 1) = ncoeffs_(order_-1) * in(c,i) - dcoeffs_(order_-1) * out(c,i);
172  }
173  }
174  }
175  else if (norder_ < dorder_){
176  for (c = 0; c < channels; ++c) {
177  for (i = 0; i < size; ++i){
178  out(c,i) = ncoeffs_(0) * in(c,i) + state_(c,0);
179  for (j = 0; j < norder_ - 1; j++)
180  {
181  state_(c,j) = ncoeffs_(j+1) * in(c,i) + state_(c,j+1) - dcoeffs_(j+1) * out(c,i);
182  }
183  for (j = norder_ - 1; j < stateSize - 1; j++)
184  {
185  state_(c,j) = state_(c,j+1) - dcoeffs_(j+1) * out(c,i);
186  }
187  state_(c,stateSize - 1) = -dcoeffs_(order_ - 1) * out(c,i);
188  }
189  }
190  }
191  else {
192  for (c = 0; c < channels; ++c) {
193  for (i = 0; i < size; ++i){
194  out(c,i) = ncoeffs_(0) * in(c,i) + state_(c,0);
195  for (j = 0; j < dorder_ - 1; j++)
196  {
197  state_(c,j) = ncoeffs_(j+1) * in(c,i) + state_(c,j+1) - dcoeffs_(j+1) * out(c,i);
198  }
199  for (j = dorder_ - 1; j < stateSize - 1; j++)
200  {
201  state_(c,j) = ncoeffs_(j+1) * in(c,i) + state_(c,j+1);
202  }
203  state_(c,stateSize - 1) = ncoeffs_(order_-1) * in(c,i);
204  }
205  }
206  }
207  out *= gain;
