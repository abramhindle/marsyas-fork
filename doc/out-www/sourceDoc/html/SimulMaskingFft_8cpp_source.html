34 static inline mrs_real IntPow (mrs_real a, mrs_natural b)
35 {
36  if (b == 0)
37  return 1.;
38  mrs_real dResult = a;
39  while (--b > 0)
40  dResult *= a;
41  return (dResult < 1e-30)? 0 : dResult;
42 }

145 SimulMaskingFft::myProcess(realvec& in, realvec& out)
146 {
147  for (mrs_natural t = 0; t < inSamples_; t++)
148  {
149  in.getCol(t, processBuff_);
150 
151  // normalize spectrum
152  processBuff_ *= normFactor_;
153  processBuff_ *= processBuff_;
154 
155  // weight with outer ear transfer function
156  processBuff_ *= outerEar_;
157 
158  // compute bark spectrum
159  GetBandLevels (freqBounds_, barkSpec_, false);
160 
161  // add internal noise
162  barkSpec_ += intNoise_;
163 
164  // compute spreading function
165  CalcSpreading (barkSpec_, excPattern_);
166 
167  // apply masking threshold
168  excPattern_ *= maskingThresh_;
169 
170  // normalize input spectrum
171  in.getCol(t, processBuff_);
172  processBuff_ *= normFactor_;
173  processBuff_ *= processBuff_;
174 
175  // compute difference
176  ComputeDifference (out, processBuff_, t);
177 
178  }

183 SimulMaskingFft::ComputeDifference (mrs_realvec &out, mrs_realvec &in, mrs_natural t)
184 {
185  mrs_natural i;
186  t = 0;
187 
188  for (i = 0; i < inObservations_; ++i)
189  out(i,t) = 0;
190 
191  for (mrs_natural k = 0; k < numBands_; k++)
192  {
193  mrs_real fLowFrac = freqBounds_[k].fLowFreqBound/audiosrate_ * (inObservations_<<1),
194  fHighFrac = freqBounds_[k].fUpFreqBound/audiosrate_ *(inObservations_<<1);
195  mrs_natural iLowBin = (mrs_natural)ceil (fLowFrac),
196  iHighBin = (mrs_natural)floor(fHighFrac);
197  for (i = iLowBin; i <= iHighBin; ++i)
198  {
199  if (excPattern_(k) <= 1./truncTresh*in(i))
200  out(i,t) = truncTresh;
201  else if (excPattern_(k) >= truncTresh*in(i))
202  out(i,t) = 1./truncTresh;
203  else
204  out(i,t) = in(i) / excPattern_(k);
205  }
206  }

221 SimulMaskingFft::GetBandLevels (FrequencyBands_t *pFrequencyValues, mrs_realvec &bandLevels, mrs_bool bDezibel)
222 {
223 
224  for (mrs_natural i = 0; i < numBands_; ++i)
225  {
226  mrs_real fLowFrac = pFrequencyValues[i].fLowFreqBound/audiosrate_ * (inObservations_<<1),
227  fHighFrac = pFrequencyValues[i].fUpFreqBound/audiosrate_ *(inObservations_<<1);
228  mrs_natural iLowBin = (mrs_natural)ceil (fLowFrac),
229  iHighBin = (mrs_natural)floor(fHighFrac);
230 
231  fLowFrac = iLowBin - fLowFrac;
232  fHighFrac = fHighFrac - iHighBin;
233  bandLevels(i) = fLowFrac * processBuff_(max(0L,iLowBin-1));
234  bandLevels(i) += fHighFrac * processBuff_(min((mrs_natural)(inObservations_ - .5),iHighBin+1));
235  for (mrs_natural j = iLowBin; j < iHighBin; j++)
236  bandLevels(i) += processBuff_(j);
237  if (bDezibel)
238  {
239  bandLevels(i) = max (bandLevels(i), 1e-20);
240  bandLevels(i) = 10./log(10.) * log ((bandLevels(i)));
241  }
242  }
243 
244  return;

247 void
248 SimulMaskingFft::CalcSpreading (mrs_realvec &bandLevels, mrs_realvec &result)
249 {
250  // this is level dependent adapted from ITU-R BS.1387
251 
252  mrs_natural iBarkj, // Masker
253  iBarkk; // Maskee
254 
255  mrs_real fTmp1,
256  fTmp2,
257  fSlope,
258  fScale = sqrt(8./3.),
259  fBRes = barkRes_,//hertz2bark (.5*audiosrate_, h2bIdx)/numBands_,
260  *pfEnPowTmp = processBuff_.getData (),
261  *pfSlopeUp = helpBuff_.getData ();
262  mrs_real *pfSlope = slopeSpread_.getData (),
263  *pfNorm = normSpread_.getData ();
264 
265  // initialize pfResult
266  result.setval(0);
267 
268  fSlope = exp ( -fBRes * 2.7 * 2.302585092994045684017991454684364207601101488628772976033);
269  fTmp2 = 1.0 / (1.0 - fSlope);
270  for (iBarkj = 0; iBarkj < numBands_; iBarkj++)
271  {
272  pfSlopeUp[iBarkj] = pfSlope[iBarkj] * pow (bandLevels(iBarkj)*fScale, .2 * fBRes);
273  fTmp1 = (1.0 - IntPow (fSlope, iBarkj+1)) * fTmp2;
274  fTmp2 = (1.0 - IntPow(pfSlopeUp[iBarkj], numBands_ - iBarkj)) / (1.0 - pfSlopeUp[iBarkj]);
275  if (bandLevels(iBarkj) < 1e-20)
276  {
277  pfSlopeUp[iBarkj] = 0;
278  pfEnPowTmp[iBarkj] = 0;
279  continue;
280  }
281  pfSlopeUp[iBarkj] = exp (0.4 * log (pfSlopeUp[iBarkj]));
282  pfEnPowTmp[iBarkj] = exp (0.4 * log (bandLevels(iBarkj)/(fTmp1 + fTmp2 -1)));
283  }
284  fSlope = exp ( 0.4 * log (fSlope));
285 
286  // lower slope
287  result(numBands_-1) = pfEnPowTmp[numBands_-1];
288  for (iBarkk = numBands_-2; iBarkk >= 0; iBarkk--)
289  result(iBarkk) = pfEnPowTmp[iBarkk] + result(iBarkk + 1) * fSlope;
290 
291  // upper slope
292  for (iBarkj = 0; iBarkj < numBands_-1; iBarkj++)
293  {
294  fSlope = pfSlopeUp[iBarkj];
295  fTmp1 = pfEnPowTmp[iBarkj];
296  for (iBarkk = iBarkj+1; iBarkk < numBands_; iBarkk++)
297  {
298  mrs_real dTmp1 = fTmp1 * fSlope;
299  fTmp1 = (dTmp1 < 1e-30)? 0 : (mrs_real)dTmp1;
300  result(iBarkk) += fTmp1;
301  }
302  }
303 
304  // normalization
305  for (iBarkk = 0; iBarkk < numBands_; iBarkk++)
306  {
307  mrs_real dTmp = result(iBarkk);
308  result(iBarkk) = sqrt(dTmp) * dTmp * dTmp *pfNorm[iBarkk];
309  //result(iBarkk) = sqrt (result(iBarkk));
310  //result(iBarkk) *= result(iBarkk)*result(iBarkk)*result(iBarkk)*result(iBarkk);
311  //result(iBarkk) *= pfNorm[iBarkk];
312  }
313  return;
314 }

316 void
317 SimulMaskingFft::ComputeTables ()
318 {
319  mrs_natural i;
320 
321 
322  // outer ear transfer function
323  {
324  for (i = 0; i < inObservations_; ++i)
325  {
326  mrs_real dTmp;
327  mrs_real fkFreq = i * .5e-3 * audiosrate_ / inObservations_;
328  if (fkFreq < 1e-10)
329  {
330  outerEar_(i) = 0;
331  continue;
332  }
333  dTmp = -.2184 * pow (fkFreq, -.8);
334  dTmp += .65 * exp (-.6 * (fkFreq-3.3)*(fkFreq-3.3));
335  //outerEar_(i) = (dTmp > 1e-37)? (mrs_real)dTmp : 0;
336  outerEar_(i) = dTmp - 1e-4 * pow (fkFreq, 3.6);
337  if (outerEar_(i) < -12)
338  outerEar_(i) = 0;
339  else
340  outerEar_(i) = pow (10.,outerEar_(i));
341  }
342  }
343 
344  // frequency bands and spreading
345  {
346  mrs_real fLowBark = hertz2bark (lowFreq, h2bIdx);
347  mrs_real fMaxBark = hertz2bark (.5*audiosrate_-1, h2bIdx);
348  for (i = 0; i < numBands_; ++i)
349  {
350  freqBounds_[i].fLowBarkBound = min(fMaxBark,fLowBark + i*barkRes_);
351  freqBounds_[i].fMidBark = min(fMaxBark,freqBounds_[i].fLowBarkBound + .5*barkRes_);
352  freqBounds_[i].fUpBarkBound = min(fMaxBark,freqBounds_[i].fLowBarkBound + barkRes_);
353 
354  freqBounds_[i].fLowFreqBound = bark2hertz (freqBounds_[i].fLowBarkBound, h2bIdx);
355  freqBounds_[i].fMidFreq = bark2hertz (freqBounds_[i].fMidBark, h2bIdx);
356  freqBounds_[i].fUpFreqBound = bark2hertz (freqBounds_[i].fUpBarkBound, h2bIdx);
357  }
358 
359  for (i = 0; i < numBands_; ++i)
360  {
361  mrs_natural cBarkk;
362  mrs_real fAtt = 1.0,
363  fNorm = 1.0,
364  fSlope = pow (10.0, -2.7 * barkRes_);
365 
366  slopeSpread_(i) = 24.0 + 230./freqBounds_[i].fMidFreq;
367  slopeSpread_(i) = pow (10.,-0.1 * barkRes_ * slopeSpread_(i));
368 
369  processBuff_(i) = 1.0;
370  // lower slope
371  for (cBarkk = i; cBarkk > 0; cBarkk--)
372  {
373  mrs_real dTmp = fAtt * fSlope;
374  fAtt = (dTmp < 1e-30) ? 0 : (mrs_real) dTmp;
375 
376  processBuff_(cBarkk-1) = fAtt; // nonnormalized masking threshold
377  fNorm += fAtt; // normalization factor
378  }
379 
380 
381  // initialize new
382  fAtt = 1.0F;
383 
384  // upper slope
385  for (cBarkk = i; cBarkk < numBands_-1; cBarkk++)
386  {
387  mrs_real dTmp = fAtt * slopeSpread_(i);
388  fAtt = (dTmp < 1e-30) ? 0 : (mrs_real) dTmp;
389 
390  processBuff_(cBarkk+1) = fAtt;
391  fNorm += fAtt; // normalization factor
392  }
393 
394  fNorm = 1.0/fNorm;
395 
396  // nonlinear superposition
397  for (cBarkk = 0; cBarkk < numBands_; cBarkk++)
398  {
399  processBuff_(cBarkk) *= fNorm;
400  normSpread_(cBarkk) += pow (processBuff_(cBarkk), 0.4);
401  }
402  }
403  for (i = 0; i < numBands_; ++i)
404  normSpread_(i) = pow (normSpread_(i), -2.5); // resulting normalization pattern (eq. 19)
405  }
406 
407  {
408  // masking (eq.25)
409  for ( i = 0; i <numBands_; ++i )
410  {
411  intNoise_(i) = .1456 * pow (.001 * freqBounds_[i].fMidFreq, -0.8); // noise in dB
412  intNoise_(i) = pow (10., intNoise_(i)); // -> in energy domain
413  }
414  }
415 
416  {
417  // masking (eq.25)
418  mrs_real v = pow (.1, .3);
419  for ( i = 0; i < 12.0/barkRes_; ++i )
420  maskingThresh_(i) = v;
421 
422 
423  while (i < numBands_)
424  {
425  maskingThresh_(i) = pow (.1, .025 * barkRes_ * i);
426  ++i;
427  }
428  }
429 
430  return;
431 }
