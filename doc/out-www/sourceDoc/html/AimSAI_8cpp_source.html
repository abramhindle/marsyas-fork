257 void
258 AimSAI::myProcess(realvec& in, realvec& out)
259 {
260  // cout << "AimSAI::myProcess" << endl;
261  mrs_natural _max_concurrent_strobes = ctrl_max_concurrent_strobes_->to<mrs_natural>();
262  mrs_real _israte = ctrl_israte_->to<mrs_real>();
263  // mrs_natural _inSamples = ctrl_inSamples_->to<mrs_natural>();
264 
265  // Grab the centre frequencies from the input
266  for (int o = 0; o < channel_count_; ++o) {
267  centre_frequencies_[o] = in(o + channel_count_);
268  // cout << "centre_frequencies_[" << o << "]=" << centre_frequencies_[o] << endl;
269  }
270 
271  findStrobes(in);
272 
273  // Reset the next strobe times
274  next_strobes_.clear();
275  next_strobes_.resize(channel_count_, 0);
276 
277  // Offset the times on the strobes from the previous buffer
278  for (int o = 0; o < channel_count_; ++o) {
279  active_strobes_[o].ShiftStrobes(ctrl_inSamples_->to<mrs_natural>());
280  }
281 
282  // Loop over samples to make the SAI
283  for (int t = 0; t < ctrl_inSamples_->to<mrs_natural>(); ++t)
284  {
285  double decay_factor = pow(sai_decay_factor_, fire_counter_);
286 
287  // Loop over channels
288  for (int o = 0; o < channel_count_; ++o)
289  {
290  // cout << "o=" << o << endl;
291  // cout << "\tch=" << o << endl;
292  // Local convenience variables
293  StrobeList &active_strobes = active_strobes_[o];
294  unsigned int next_strobe_index = next_strobes_[o];
295 
296  // Update strobes
297  // If we are up to or beyond the next strobe...
298  if (next_strobe_index < strobes_[o].size()) {
299  // cout << "(next_strobe_index < strobes_[o].size())" << endl;
300  // cout << "\t\tnext_strobe_index" << next_strobe_index << endl;
301  // cout << "\t\tinput.strobe(ch, next_strobe_index)=" << strobes_[o][next_strobe_index] << endl;
302 
303  if (t == strobes_[o][next_strobe_index]) {
304  // cout << "(t == strobes_[o][next_strobe_index])" << endl;
305  // A new strobe has arrived.
306  // If there are too many strobes active, then get rid of the
307  // earliest one
308  if (active_strobes.strobe_count() >= _max_concurrent_strobes) {
309  active_strobes.DeleteFirstStrobe();
310  }
311 
312  // Add the active strobe to the list of current strobes and
313  // calculate the strobe weight
314  double weight = 1.0;
315  if (active_strobes.strobe_count() > 0) {
316  int last_strobe_time = active_strobes.Strobe(
317  active_strobes.strobe_count() - 1).time;
318 
319  // If the strobe occured within 10 impulse-response
320  // cycles of the previous strobe, then lower its weight
321  // cout << "t=" << t << "\tlast_strobe_time=" << last_strobe_time << "_israte=" << _israte << " cf=" << centre_frequencies_[o] << endl;
322  weight = (t - last_strobe_time) / _israte
323  * centre_frequencies_[o] / 10.0;
324  if (weight > 1.0)
325  weight = 1.0;
326  }
327  active_strobes.AddStrobe(t, weight);
328  next_strobe_index++;
329 
330  // Update the strobe weights
331  double total_strobe_weight = 0.0;
332  for (int si = 0; si < active_strobes.strobe_count(); ++si) {
333  // cout << "si=" << si << "\tw=" << active_strobes.Strobe(si).weight << "\tw2=" << strobe_weights_[active_strobes.strobe_count() - si - 1] << endl;
334  total_strobe_weight += (active_strobes.Strobe(si).weight
335  * strobe_weights_[active_strobes.strobe_count() - si - 1]);
336  }
337  for (int si = 0; si < active_strobes.strobe_count(); ++si) {
338  // cout << "si=" << si << "\tw1=" << active_strobes.Strobe(si).weight << "\tw2=" << strobe_weights_[active_strobes.strobe_count() - si - 1] << " tot=" << total_strobe_weight << endl;
339  active_strobes.SetWorkingWeight(si,
340  (active_strobes.Strobe(si).weight
341  * strobe_weights_[active_strobes.strobe_count() - si - 1])
342  / total_strobe_weight);
343  }
344  }
345  }
346 
347  // Remove inactive strobes
348  while (active_strobes.strobe_count() > 0) {
349  // Get the relative time of the first strobe, and see if it exceeds
350  // the maximum allowed time.
351  if ((t - active_strobes.Strobe(0).time) > max_strobe_delay_idx_) {
352  active_strobes.DeleteFirstStrobe();
353  } else {
354  // cout << "break" << endl;
355  break;
356  }
357  }
358 
359  // Update the SAI buffer with the weighted effect of all the active
360  // strobes at the current sample
361  for (int si = 0; si < active_strobes.strobe_count(); ++si) {
362  // cout << "si=" << si << endl;
363  // Add the effect of active strobe at correct place in the SAI buffer
364  // Calculate 'delay', the time from the strobe event to now
365  int delay = t - active_strobes.Strobe(si).time;
366 
367  // If the delay is greater than the (user-set)
368  // minimum strobe delay, the strobe can be used
369  if (delay >= min_strobe_delay_idx_ && delay < max_strobe_delay_idx_) {
370  // The value at be added to the SAI
371  double sig = in(o, t);
372  // cout << "sig=" << sig << "\tww=" << active_strobes.Strobe(si).working_weight << "\tdecay=" << decay_factor << endl;
373 
374  // Weight the sample correctly
375  sig *= active_strobes.Strobe(si).working_weight;
376 
377  // Adjust the weight acording to the number of samples until the
378  // next output frame
379  sig *= decay_factor;
380 
381  // Update the temporary SAI buffer
382  // cout << "output1(" << o << "," << delay << ")=" << out(o, delay) + sig << endl;
383  out(o, delay) = out(o, delay) + sig;
384  }
385  }
386 
387  next_strobes_[o] = next_strobe_index;
388  } // End loop over channels
389 
390  fire_counter_--;
391 
392  // cout << "fire_counter_=" << fire_counter_ << endl;
393 
394  // Check to see if we need to output an SAI frame on this sample
395  if (fire_counter_ <= 0) {
396  // cout << "(fire_counter_ <= 0)" << endl;
397  // Decay the SAI by the correct amount and add the current output frame
398  double decay = pow(sai_decay_factor_, frame_period_samples_);
399 
400  for (int o = 0; o < channel_count_; ++o) {
401  // for (int t = 0; t < _inSamples; ++t ) {
402  for (int t = 0; t < frame_period_samples_; ++t ) {
403  // cout << "output2(" << o << "," << t << ")=" << sai_temp_(o,t) + out(o,t) * decay << endl;
404  out(o, t) = sai_temp_(o,t) + out(o,t) * decay;
405  }
406  }
407 
408  // Zero the temporary signal
409  for (int o = 0; o < sai_temp_.getRows(); ++o) {
410  for (int t = 0; t < sai_temp_.getCols(); ++t) {
411  sai_temp_(o, t) = 0.0;
412  }
413  }
414  fire_counter_ = frame_period_samples_ - 1;
415  }
416  } // End loop over samples
417

419  // for (int t = 0; t < ctrl_inSamples_->to<mrs_natural>(); ++t) {
420  // for (int o = channel_count_; o < channel_count_ * 2; ++o) {
421  // out(o,t) = in(o,t);
