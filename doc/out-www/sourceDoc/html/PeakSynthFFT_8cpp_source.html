82 void
83 PeakSynthFFT::generateMask(mrs_natural type)
84 {
85  mrs_natural i, j;
86  realvec peaks = ctrl_Peaks_->to<mrs_realvec>();
87 
88  //cout << peaks;
89  mrs_natural nbPeaks = peaks.getSize()/peakView::nbPkParameters;
90 
91  // intialize background
92  if(bgVolume_ != -1)
93  {
94  mask_.setval(bgVolume_);
95  if(type == 1)
96  mask_*=(1-bgPanning_)/2;
97  if(type == 2)
98  mask_*=(1+bgPanning_)/2;
99  }
100  else
101  mask_.setval(0);
102 
103  // set level info for foreground clusters
104  for (i=0 ; i<onObservations_/2 + 1; ++i)
105  {
106  for(j=0 ; j<nbPeaks; j++)
107  {
108  if(peaks(j+peakView::pkGroup*nbPeaks) <= -1)
109  continue;
110  if (i>=peaks(j+nbPeaks*peakView::pkBinLow)*onObservations_ && i<=peaks(j+nbPeaks*peakView::pkBinHigh)*onObservations_)
111  {
112  mrs_real vol, pan;
113  if(fgVolume_ != -1)
114  {
115  // use global info
116  vol = fgVolume_;
117  pan = fgPanning_;
118  }
119  else
120  {
121  // use peaks info
122  vol = peaks(j+nbPeaks*peakView::pkVolume);
123  pan = peaks(j+nbPeaks*peakView::pkPan);
124  }
125 
126  mask_(i) = vol;
127  // left
128  if(type == 1)
129  mask_(i) *= (1-pan)/2;
130  // right
131  if(type == 2)
132  mask_(i) *= (1+pan)/2;
133  break;
134  }
135  }
136  }
137 // cout << mask_;
138 }

140 void
141 PeakSynthFFT::lpfMask()
142 {
143  mrs_real gain = 0.8, deltaGain=.3;
144 
145  for (mrs_natural i=0 ; i<mask_.getSize() ; ++i)
146  {
147  mrs_real g=gain-deltaGain*(mask_.getSize()-i)/mask_.getSize();
148  mask_(i) = g*mask_(i)+(1-g)*lastMask_(i);
149  }
150  lastMask_ = mask_;
151 }

153 void
154 PeakSynthFFT::myProcess(realvec& in, realvec& out)
155 {
156  mrs_natural nbChannels = ctrl_NbChannels_->to<mrs_natural>();
157  mrs_natural t,o;
158  //cout << ctrl_Peaks_->to<mrs_realvec>();
159  for (t = 0; t < onSamples_; t++)
160  {
161  generateMask(t+(nbChannels-1));
162 
163 #ifdef MARSYAS_MATLAB
164 #ifdef MTLB_DBG_LOG
165  MATLAB_PUT(mask_, "mask");
166  MATLAB_EVAL("figure(71),plot(mask),axis('tight'),grid on");
167 #endif
168 #endif
169  lpfMask();
170  for (o=0; o < onObservations_/2+1; o++)
171  {
172  //apply PeakSynthFFT to all channels
173  out(o,t) = mask_(o) * in(o, 0);
174  }
175  for (o=onObservations_/2+1; o < onObservations_; o++)
176  {
177  //apply PeakSynthFFT to all channels
178  out(o,t) = in(o, 0);
179  }
180  }
181 // cout << out;
182 }
