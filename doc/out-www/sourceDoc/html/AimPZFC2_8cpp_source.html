233 bool
234 AimPZFC2::SetPZBankCoeffs() {
239  if (ctrl_use_fit_->to<mrs_bool>()) {
240  if (!SetPZBankCoeffsERBFitted())
241  return false;
242  } else {
243  if (!SetPZBankCoeffsOrig())
244  return false;
245  }
246 
247  double mindamp = ctrl_mindamp_->to<mrs_real>();
248  double maxdamp = ctrl_maxdamp_->to<mrs_real>();
249 
250  rmin_.resize(channel_count_);
251  rmax_.resize(channel_count_);
252  xmin_.resize(channel_count_);
253  xmax_.resize(channel_count_);
254 
255  for (int c = 0; c < channel_count_; ++c) {
256  // Calculate maximum and minimum damping options
257  rmin_[c] = exp(-mindamp * pole_frequencies_(c));
258  rmax_[c] = exp(-maxdamp * pole_frequencies_(c));
259 
260  xmin_[c] = rmin_[c] * cos(pole_frequencies_(c)
261  * pow((1-pow(mindamp, 2)), 0.5));
262  xmax_[c] = rmax_[c] * cos(pole_frequencies_(c)
263  * pow((1-pow(maxdamp, 2)), 0.5));
264 
265  }
266 
267  // Set up AGC parameters
268  agc_stage_count_ = 4;
269  agc_epsilons_.create(agc_stage_count_);
270  agc_epsilons_(0) = 0.0064;
271  agc_epsilons_(1) = 0.0016;
272  agc_epsilons_(2) = 0.0004;
273  agc_epsilons_(3) = 0.0001;
274 
275  agc_gains_.create(agc_stage_count_);
276  agc_gains_(0) = 1.0;
277  agc_gains_(1) = 1.4;
278  agc_gains_(2) = 2.0;
279  agc_gains_(3) = 2.8;
280 
281  double mean_agc_gain = 0.0;
282  for (int c = 0; c < agc_stage_count_; ++c)
283  mean_agc_gain += agc_gains_(c);
284  mean_agc_gain /= static_cast<double>(agc_stage_count_);
285 
286  for (int c = 0; c < agc_stage_count_; ++c)
287  agc_gains_(c) /= mean_agc_gain;
288 
289  return true;
290 }

292 bool
293 AimPZFC2::SetPZBankCoeffsOrig() {
294  // This function sets the following variables:
295  // channel_count_
296  // pole_dampings_
297  // pole_frequencies_
298  // za0_, za1_, za2
299  // output_
300 
301  double sample_rate = getctrl("mrs_real/israte")->to<mrs_real>();
302  double cf_max = getctrl("mrs_real/cf_max")->to<mrs_real>();
303  double cf_min = getctrl("mrs_real/cf_min")->to<mrs_real>();
304  double bandwidth_over_cf = getctrl("mrs_real/bandwidth_over_cf")->to<mrs_real>();
305  double min_bandwidth_hz = getctrl("mrs_real/min_bandwidth_hz")->to<mrs_real>();
306  double step_factor = getctrl("mrs_real/step_factor")->to<mrs_real>();
307  double pole_damping = getctrl("mrs_real/pole_damping")->to<mrs_real>();
308  double zero_factor = getctrl("mrs_real/zero_factor")->to<mrs_real>();
309  double zero_damping = getctrl("mrs_real/zero_damping")->to<mrs_real>();
310 
311  // TODO(tomwalters): There's significant code-duplication between this function
312  // and SetPZBankCoeffsERBFitted, and SetPZBankCoeffs
313 
314  // Normalised maximum pole frequency
315  double pole_frequency = cf_max / sample_rate * (2.0 * PI);
316  channel_count_ = 0;
317  while ((pole_frequency / (2.0 * PI)) * sample_rate > cf_min) {
318  double bw = bandwidth_over_cf * pole_frequency + 2 * PI * min_bandwidth_hz / sample_rate;
319  pole_frequency -= step_factor * bw;
320  channel_count_++;
321  }
322 
323 
324  // Now the number of channels is known, various buffers for the filterbank
325  // coefficients can be initialised
326 
327 
328 
329 
330  pole_dampings_.stretch(channel_count_);
331  pole_dampings_.setval(pole_damping);
332 
333 
334  pole_frequencies_.stretch(channel_count_);
335  pole_frequencies_.setval(0.0);
336 
337 
338 
339  // Direct-form coefficients
340  za0_.clear();
341  za0_.resize(channel_count_, 0.0);
342  za1_.clear();
343  za1_.resize(channel_count_, 0.0);
344  za2_.clear();
345  za2_.resize(channel_count_, 0.0);
346 
347  // The output signal bank
348  // output_.Initialize(channel_count_, buffer_length_, sample_rate);
349 
350  // cout.precision(20);
351 
352  // cout << "********** 1/3=" << 1.0/3.0 << endl;
353 
354  // Reset the pole frequency to maximum
355  pole_frequency = cf_max / sample_rate * (2.0 * PI);
356  // cout << "cf_max=" << cf_max << endl;
357  // cout << "sample_rate=" << sample_rate << endl;
358  // cout << "pole_frequency=" << pole_frequency << endl;
359 
360  centre_frequencies_.clear();
361  centre_frequencies_.resize(channel_count_);
362 
363  for (int i = channel_count_ - 1; i > -1; --i) {
364  // cout << "i=" << i << endl;
365 
366  // Store the normalised pole frequncy
367  pole_frequencies_(i) = pole_frequency;
368 
369  // Calculate the real pole frequency from the normalised pole frequency
370  double frequency = pole_frequency / (2.0 * PI) * sample_rate;
371 
372  // Store the real pole frequency as the 'centre frequency' of the filterbank
373  // channel
374  centre_frequencies_[i] = frequency;
375  // output_.set_centre_frequency(i, frequency);
376 
377  double zero_frequency = Minimum(PI, zero_factor * pole_frequency);
378  // cout << "\tzero_frequency=" << zero_frequency << endl;
379 
380  // Impulse-invariance mapping
381  double z_plane_theta = zero_frequency * sqrt(1.0 - pow(zero_damping, 2));
382  double z_plane_rho = exp(-zero_damping * zero_frequency);
383  // cout << "\tz_plane_theta=" << z_plane_theta << endl;
384  // cout << "\tz_plane_rho=" << z_plane_rho << endl;
385  // Direct-form coefficients from z-plane rho and theta
386  double a1 = -2.0 * z_plane_rho * cos(z_plane_theta);
387  double a2 = z_plane_rho * z_plane_rho;
388 
389  // Normalised to unity gain at DC
390  double a_sum = 1.0 + a1 + a2;
391  // cout << "\ta1=" << a1 << endl;
392  // cout << "\ta2=" << a2 << endl;
393  // cout << "\ta_sum=" << a_sum << endl;
394 
395  za0_[i] = 1.0 / a_sum;
396  za1_[i] = a1 / a_sum;
397  za2_[i] = a2 / a_sum;
398 
399  // Subtract step factor (1/n2) times current bandwidth from the pole
400  // frequency
401  double bw = bandwidth_over_cf * pole_frequency + 2 * PI * min_bandwidth_hz / sample_rate;
402  // cout << "\tmin_bandwidth_hz_=" << min_bandwidth_hz << endl;
403  // cout << "\tbw=" << bw << endl;
404  // cout << "\tstep_factor=" << step_factor << endl;
405  pole_frequency -= step_factor * bw;
406  }
407 
408  return true;
409 }

499 bool
500 AimPZFC2::SetPZBankCoeffsERBFitted() {
501  // cout << "ModulePZFC::SetPZBankCoeffsERBFitted" << endl;
502  // cout << "AimPZFC2::SetPZBankCoeffsERBFitted" << endl;
503  //double parameter_values[3 * 7] = {
505  //1.14827, 0.00000, 0.00000, // % SumSqrErr= 10125.41
506  //0.53571, -0.70128, 0.63246, // % RMSErr = 2.81586
507  //0.76779, 0.00000, 0.00000, // % MeanErr = 0.00000
509  //0.00000, 0.00000, 0.00000,
510  //6.00000, 0.00000, 0.00000,
511  //1.08869, -0.09470, 0.07844,
512  //10.56432, 2.52732, 1.86895
514  //};
515 
516  double parameter_values[3 * 7] = {
517  // Fit 515 from Dick
518  // Final, Nfit = 515, 9-3 parameters, PZFC, cwt 0
519  1.72861, 0.00000, 0.00000, // SumSqrErr = 13622.24
520  0.56657, -0.93911, 0.89163, // RMSErr = 3.26610
521  0.39469, 0.00000, 0.00000, // MeanErr = 0.00000
522  // Inf, 0.00000, 0.00000, // RMSCost = NaN - would set coefc to infinity, but this isn't passed on
523  0.00000, 0.00000, 0.00000,
524  2.00000, 0.00000, 0.00000, //
525  1.27393, 0.00000, 0.00000,
526  11.46247, 5.46894, 0.11800
527  // -4.15525, 1.54874, 2.99858 // Kv
528  };
529 
530  double sample_rate = getctrl("mrs_real/israte")->to<mrs_real>();
531  double cf_max = getctrl("mrs_real/cf_max")->to<mrs_real>();
532  double cf_min = getctrl("mrs_real/cf_min")->to<mrs_real>();
533 
534  // Precalculate the number of channels required - this method is ugly but it
535  // was the quickest way of converting from MATLAB as the step factor between
536  // channels can vary quadratically with pole frequency...
537 
538  // Normalised maximum pole frequency
539  double pole_frequency = cf_max / sample_rate * (2.0 * PI);
540 
541  channel_count_ = 0;
542  while ((pole_frequency / (2.0 * PI)) * sample_rate > cf_min) {
543  double frequency = pole_frequency / (2.0 * PI) * sample_rate;
544  double f_dep = ERBTools::Freq2ERB(frequency)
545  / ERBTools::Freq2ERB(1000.0) - 1.0;
546  double bw = ERBTools::Freq2ERBw(pole_frequency
547  / (2.0 * PI) * sample_rate);
548  double step_factor = 1.0
549  / (parameter_values[4*3] + parameter_values[4 * 3 + 1]
550  * f_dep + parameter_values[4 * 3 + 2] * f_dep * f_dep); // 1/n2
551  pole_frequency -= step_factor * (bw * (2.0 * PI) / sample_rate);
552  channel_count_++;
553  }
554 
555  // Now the number of channels is known, various buffers for the filterbank
556  // coefficients can be initialised
557 
558 
559  cout << "channel_count_ = " << channel_count_ << endl;
560  pole_dampings_.stretch(channel_count_);
561  pole_dampings_.setval(0.0);
562  cout << pole_dampings_ << endl;
563 
564 
565  pole_frequencies_.stretch(channel_count_);
566  pole_frequencies_.setval(0.0);
567 
568  // Direct-form coefficients
569  za0_.clear();
570  za0_.resize(channel_count_, 0.0);
571  za1_.clear();
572  za1_.resize(channel_count_, 0.0);
573  za2_.clear();
574  za2_.resize(channel_count_, 0.0);
