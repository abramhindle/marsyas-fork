18  /**************************************************************
19  * Given an N x N matrix A, this routine replaces it by the LU *
20  * decomposition of a rowwise permutation of itself. A and N *
21  * are input. INDX is an output vector which records the row *
22  * permutation effected by the partial pivoting; D is output *
23  * as -1 or 1, depending on whether the number of row inter- *
24  * changes was even or odd, respectively. This routine is used *
25  * in combination with LUBKSB to solve linear equations or to *
26  * invert a matrix. Return code is 1, if matrix is singular. *
27  **************************************************************/
28 int LUDCMP(REAL **A, int n, int *INDX, int *d) {
29 
30  REAL AMAX,DUM, SUM;
31  int I,IMAX,J,K;
32  IMAX = 0;
33 
34  REAL *VV;
35  void *vmblock = NULL;
36 
37  vmblock = vminit();
38  VV = (REAL *) vmalloc(vmblock, VEKTOR, NMAX, 0);
39 
40  if (! vmcomplete(vmblock))
41  {
42  // LogError ("No Memory", 0, __FILE__, __LINE__);
43  return -1;
44  }
45 
46  *d=1;
47 
48  for (I=1; I<n+1; I++) {
49  AMAX=0.0;
50  for (J=1; J<n+1; J++)
51  if (ABS(A[I][J]) > AMAX) AMAX=ABS(A[I][J]);
52 
53  if(AMAX < TINY)
54  return 1; //singular Matrix!
55  VV[I] = 1.0 / AMAX;
56  } // i loop
57 
58  for (J=1; J<n+1;J++) {
59 
60  for (I=1; I<J; I++) {
61  SUM = A[I][J];
62  for (K=1; K<I; K++)
63  SUM = SUM - A[I][K]*A[K][J];
64  A[I][J] = SUM;
65  } // i loop
66  AMAX = 0.0;
67 
68  for (I=J; I<n+1; I++) {
69  SUM = A[I][J];
70  for (K=1; K<J; K++)
71  SUM = SUM - A[I][K]*A[K][J];
72  A[I][J] = SUM;
73  DUM = VV[I]*ABS(SUM);
74  if (DUM >= AMAX) {
75  IMAX = I;
76  AMAX = DUM;
77  }
78  } // i loop
79 
80  if (J != IMAX) {
81  for (K=1; K<n+1; K++) {
82  DUM = A[IMAX][K];
83  A[IMAX][K] = A[J][K];
84  A[J][K] = DUM;
85  } // k loop
86  *d = -*d;
87  VV[IMAX] = VV[J];
88  }
89 
90  INDX[J] = IMAX;
91 
92  if (ABS(A[J][J]) < TINY) A[J][J] = TINY;
93 
94  if (J != n) {
95  DUM = 1.0 / A[J][J];
96  for (I=J+1; I<n+1; I++)
97  A[I][J] *= DUM;
98  }
99  } // j loop
100 
101  free(vmblock);
102  return 0;
103 
104 } // subroutine LUDCMP

107  /*****************************************************************
108  * Solves the set of N linear equations A . X = B. Here A is *
109  * input, not as the matrix A but rather as its LU decomposition, *
110  * determined by the routine LUDCMP. INDX is input as the permuta-*
111  * tion vector returned by LUDCMP. B is input as the right-hand *
112  * side vector B, and returns with the solution vector X. A, N and*
113  * INDX are not modified by this routine and can be used for suc- *
114  * cessive calls with different right-hand sides. This routine is *
115  * also efficient for plain matrix inversion. *
116  *****************************************************************/
117 void LUBKSB(REAL **A, int n, int *INDX, REAL *B) {
118  REAL SUM;
119  int I,II,J,LL;
120 
121  II = 0;
122 
123  for (I=1; I<n+1; I++) {
124  LL = INDX[I];
125  SUM = B[LL];
126  B[LL] = B[I];
127  if (II != 0)
128  for (J=II; J<I; J++)
129  SUM = SUM - A[I][J]*B[J];
130  else if (SUM != 0.0) II = I;
131  B[I] = SUM;
132  } // i loop
133 
134  for (I=n; I>0; I--) {
135  SUM = B[I];
136  if (I < n) {
137  for (J=I+1; J<n+1; J++)
138  SUM = SUM - A[I][J]*B[J];
139  }
140  B[I] = SUM / A[I][I];
141  } // i loop

144 } // LUBKSB
