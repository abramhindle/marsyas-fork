556 void
557 Marx2DGraph::plotpolyinterpolation(QPainter* painter)
558 {
559 
560  // divided difference algorithm
561  // modified from kyle louden's "algorithms in c"
562  float lastw=-1, lasth;
563  for (int h=0; h<buffersize-3; h++) {
564  float *table, *coeff, *x, *pz, *z;
565  float term;
566 
567  int steps = 64;
568 
569  table = new float[4];
570  x = new float[4];
571  coeff = new float[4];
572  pz = new float[steps];
573  z = new float[steps];
574 
575  x[0] = h+0; x[1] = h+1; x[2] = h+2; x[3] = h+3;
576  table[0] = (float)((*buffer)(h));
577  table[1] = (float)((*buffer)(h+1));
578  table[2] = (float)((*buffer)(h+2));
579  table[3] = (float)((*buffer)(h+3));
580 
581  coeff[0] = table[0];
582 
583  int j;
584  int n=4;
585  for (int k=1; k<n; k++) {
586  for (int i=0; i<n-k; i++) {
587  j = i + k;
588  table[i] = (table[i+1] - table[i])/(x[j] - x[i]);
589  }
590  coeff[k] = table[0];
591  }
592 
593  int m=steps;
594  float zn=h;
595  for (int k=0; k<m; k++) {
596  pz[k] = coeff[0];
597 
598  for (j=1; j<n; j++) {
599  term = coeff[j];
600 
601  for (int i=0; i<j; i++) {
602  term = term * (zn - x[i]);
603  }
604 
605  pz[k] = pz[k] + term;
606  }
607 
608  z[k] = zn;
609  zn = zn + 4./m; // there are four points covered in m steps
610 
611  if ((h != buffersize-4 && k <= m/4) ||
612  h == buffersize-4) {
613  float w = (gwidth/(buffersize-1))*z[k] + xpos + xaxisoffset;
614  float h = (gheight/2.0 + ypos + yaxisoffset) - (gheight/2)*pz[k];
615  if ( w > gwidth + xpos + xaxisoffset) { // hack that needs fixing
616  break;
617  }
618 
619  if (lastw >=0) { painter->drawLine( QLineF( lastw, lasth, w, h ) ); }
620 
621  lastw = w;
622  lasth = h;
623  }
624  }
625  }

664 void
665 Marx2DGraph::plotpoints(QPainter* painter)
666 {
667  float p = point_size/2.;
668  for (int i=0; i<buffersize; i++) {
669  float val = (float)((*buffer)(i));
670 
671 
672 
673  float w = (gwidth/(buffersize-1))*i + xpos + xaxisoffset;
674  float h = (gheight/2.0 + ypos + yaxisoffset) - (gheight/2.0)*val;
675 
676  switch (point_type) {
677  case CIRCLES:
678  painter->drawEllipse( QRectF( w, h, point_size, point_size ) );
679  break;
680  case SQUARES:
681  painter->drawRect( QRectF( w, h, point_size, point_size ) );
682  break;
683  case XS:
684  painter->drawLine( QLineF(w-p, h-p, w+p, h+p ) );
685  painter->drawLine( QLineF(w-p, h+p, w+p, h-p ) );
686  break;
687  default:
688  painter->drawEllipse( QRectF( w, h, point_size, point_size ) );
689  }
690  }

723  dsamplestring = "x = ";

726  if (dsample - floor(dsample) > .5) { dsample = ceil(dsample);}
727  else { dsample = floor(dsample); }
728  dvalue = (*buffer)((int)dsample);
729 
730 
731  dvalue *= 0.5;
732  dvalue += 0.5;
733  dvalue *= (ymaxval - yminval);
734  dvalue += yminval;
735 
736 
737  }
738  else if (me->button() == Qt::RightButton) {
739  dsamplestring = "x = ";
740  dvaluestring = "y = ";
741 
742  dvalue = (mousey - gheight/2. - ypos - yaxisoffset)*-2/gheight;
743  }

747  dsamplestring.append( ds.str() );
